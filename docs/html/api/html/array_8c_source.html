22EmbArray*
23embArray_create(int type)
24{
25 EmbArray *a;
26 a = (EmbArray*)malloc(sizeof(EmbArray));
27 a->type = type;
28 a->length = CHUNK_SIZE;
29 a->count = 0;
30 switch (type) {
31 case EMB_STITCH:
32 a->stitch = (EmbStitch*)malloc(CHUNK_SIZE*sizeof(EmbStitch));
33 break;
34 case EMB_THREAD:
35 a->thread = (EmbThread*)malloc(CHUNK_SIZE*sizeof(EmbThread));
36 break;
37 default:
38 a->geometry = (EmbGeometry*)malloc(CHUNK_SIZE*sizeof(EmbGeometry));
39 break;
40 }
41 return a;
42}

49int
50embArray_resize(EmbArray *a)
51{
52 if (a->count < a->length - 3) {
53 return 1;
54 }
55 a->length += CHUNK_SIZE;
56 switch (a->type) {
57 case EMB_STITCH:
58 a->stitch = (EmbStitch*)realloc(a->stitch, a->length*sizeof(EmbStitch));
59 if (!a->stitch) {
60 /* TODO: Error reporting */
61 return 0;
62 }
63 break;
64 case EMB_THREAD:
65 a->thread = (EmbThread*)realloc(a->thread, a->length*sizeof(EmbThread));
66 if (!a->thread) {
67 /* TODO: Error reporting */
68 return 0;
69 }
70 break;
71 default:
72 a->geometry = (EmbGeometry *)realloc(a->geometry, a->length*sizeof(EmbGeometry));
73 if (!a->geometry) {
74 /* TODO: Error reporting */
75 return 0;
76 }
77 break;
78 }
79
80 return 1;
81}

86void
87embArray_copy(EmbArray *dst, EmbArray *src)
88{
89 dst = embArray_create(src->type);
90 dst->length = src->length;
91 dst->count = src->count;
92 embArray_resize(dst);
93 /* BUG: Potential failure to copy path memory, only copy pointers? */
94
95 switch (dst->type) {
96 case EMB_STITCH:
97 memcpy(dst->stitch, src->stitch, sizeof(EmbStitch)*src->count);
98 break;
99 case EMB_THREAD:
100 memcpy(dst->thread, src->thread, sizeof(EmbThread)*src->count);
101 break;
102 default:
103 memcpy(dst->geometry, src->geometry, sizeof(EmbGeometry)*src->count);
104 break;
105 }
106}

302void
303embArray_free(EmbArray* a)
304{
305 int i;
306 if (!a) {
307 return;
308 }
309 switch (a->type) {
310 case EMB_STITCH:
311 free(a->stitch);
312 break;
313 case EMB_THREAD:
314 free(a->thread);
315 break;
316 default:
317 for (i = 0; i < a->count; i++) {
318 EmbGeometry g = a->geometry[i];
319 switch (a->geometry[i].type) {
320 case EMB_PATH: {
321 embArray_free(g.object.path.pointList);
322 break;
323 }
324 case EMB_POLYGON: {
325 embArray_free(g.object.polygon.pointList);
326 break;
327 }
328 case EMB_POLYLINE: {
329 embArray_free(g.object.polyline.pointList);
330 break;
331 }
332 default:
333 break;
334 }
335 }
336 free(a->geometry);
337 break;
338 }
339 free(a);
340}
