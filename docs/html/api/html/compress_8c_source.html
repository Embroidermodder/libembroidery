63void
64huffman_build_table(huffman *h)
65{
66 int bit_length, i, max_length, size;
67 max_length = 0;
68 size = 1 << h->table_width;
69 for (i = 0; i < h->table_width; i++) {
70 if (h->lengths[i] > max_length) {
71 max_length = h->lengths[i];
72 }
73 }
74 for (bit_length=1; bit_length<=h->table_width; bit_length++) {
75 int j;
76 size /= 2;
77 for (j=0; j < h->nlengths; j++) {
78 if (h->lengths[j] == bit_length) {
79 int k;
80 for (k=0; k<size; k++) {
81 h->table[h->ntable+k] = j;
82 h->ntable++;
83 }
84 }
85 }
86 }
87}

99int *huffman_lookup(huffman h, int byte_lookup)
100{

103 out[0] = h.default_value;
104 out[1] = 0;
105 return out;
106 }
107 out[0] = h.table[byte_lookup >> (16-h.table_width)];
108 out[1] = h.lengths[out[0]];

127int
128compress_get_bits(compress *c, int length)
129{
130 int i, end_pos_in_bits, start_pos_in_bytes,
131 end_pos_in_bytes, value, mask_sample_bits,
132 unused_bits, original;
133
134 end_pos_in_bits = c->bit_position + length - 1;
135 start_pos_in_bytes = c->bit_position / 8;
136 end_pos_in_bytes = end_pos_in_bits / 8;
137 value = 0;
138
139 for (i=start_pos_in_bytes; i < end_pos_in_bytes+1; i++) {
140 value <<= 8;
141 if (i > c->input_length) {
142 break;
143 }
144 value |= (c->input_data[i]) & 0xFF;
145 }
146
147 unused_bits = (7 - end_pos_in_bits) % 8;
148 mask_sample_bits = (1<<length) - 1;
149 original = (value >> unused_bits) & mask_sample_bits;
150 return original;
151}

178{

225void
226compress_load_character_huffman(compress *c)
227{
228 int count;
229 count = compress_pop(c, 9);

231 c->character_huffman.default_value = compress_pop(c, 9);

234 int i = 0;

237 compress_peek(c, 16));

239 if (h[0]==0) {
240 i += h[0];

250 i++;

313 c->bit_position += h[1];
