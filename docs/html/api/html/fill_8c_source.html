32int lindenmayer_system(L_system L, char *state, int iterations, int complete)
33{
34 /* We know that the full length of the curve has to fit within
35 * the number of stitches and we can cancel consecutive +-, -+
36 * etc.
37 *
38
39Potential reference:
40
41@book{Prusinkiewicz1996Mar,
42 author = {Prusinkiewicz, Przemyslaw and Lindenmayer, Aristid and Hanan, J. S. and Fracchia, F. D. and Fowler, D. R. and de Boer, M. J. M. and Mercer, L.},
43 title = {{The Algorithmic Beauty of Plants (The Virtual Laboratory)}},
44 year = {1996},
45 month = {Mar},
46 publisher = {Springer}
47}
48 */
49 char *new_state;
50 int j;
51
52 if (complete == 0) {
53 state[0] = L.axiom;
54 state[1] = 0;
55 lindenmayer_system(L, state, iterations, complete+1);
56 return 0;
57 }
58
59 new_state = state + MAX_STITCHES*5;
60
61 new_state[0] = 0;
62
63 /* replace letters using rules by copying to new_state */
64 for (j=0; j < (int)strlen(state); j++) {
65 if (state[j] >= 'A' && state[j] < 'F') {
66 strcat(new_state, L.rules[state[j]-'A']);
67 }
68 if (state[j] == 'F') {
69 strcat(new_state, "F");
70 }
71 if (state[j] == '+') {
72 strcat(new_state, "+");
73 }
74 if (state[j] == '-') {
75 strcat(new_state, "-");
76 }
77 }
78 memcpy(state, new_state, strlen(new_state)+1);
79
80 if (complete < iterations) {
81 lindenmayer_system(L, state, iterations, complete+1);
82 }
83 return 0;
84}

92static void
93join_short_stitches(int *points, int *n_points, int width, int tolerence)
94{
95 int found = 1;
96 while (found > 0) {
97 int i;
98 found = 0;
99 for (i=*n_points-2; i>=0; i--) {
100 int st1 = points[i+1]%width - points[i]%width;
101 int st2 = points[i+2]%width - points[i+1]%width;
102 int same_line = (points[i+1]/width == points[i]/width)
103 && (points[i+2]/width == points[i+1]/width);
104 if (st1 < tolerence && st2 < tolerence && same_line) {
105 found++;
106 break;
107 }
108 }
109 if (found) {
110 /* Remove the point. */
111 i++;
112 for (; i<*n_points; i++) {
113 points[i] = points[i+1];
114 }
115 (*n_points)--;
116 }
117 }
118}

127static int *
128threshold_method(EmbImage *image, int *n_points,
129 int subsample_width, int subsample_height, int threshold)
130{
131 int i, j;
132 int *points;
133 int height = image->height;
134 int width = image->width;
135 points = (int *)malloc((height/subsample_height)
136 *(width/subsample_width) * sizeof(int));
137 *n_points = 0;
138 for (i=0; i<height/subsample_height; i++)
139 for (j=0; j<width/subsample_width; j++) {
140 EmbColor color;
141 int index = subsample_height*i*width+subsample_width*j;
142 color.r = image->data[3*index+0];
143 color.g = image->data[3*index+1];
144 color.b = image->data[3*index+2];
145 if (color.r+color.g+color.b < threshold) {
146 points[*n_points] = index;
147 (*n_points)++;
148 }
149 }
150 return points;
151}

166greedy_algorithm(int *points, int n_points, int width, EmbReal bias)
167{
168 int i, j;
169 printf("points[0] = %d\n", points[0]);
170 printf("n_points = %d\n", n_points);
171 printf("width = %d\n", width);
172 printf("bias = %f\n", bias);
173
174 for (i=0; i<n_points-1; i++) {
175 int stor;
176 EmbReal shortest = 1.0e20;
177 int next = i+1;
178 /* Find nearest neighbour. */
179 int x1 = points[i]%width;
180 int y1 = points[i]/width;
181 for (j=i+1; j<n_points; j++) {
182 int x, y;
183 EmbReal distance;
184 x = x1 - (points[j]%width);
185 if (x*x > shortest) {
186 continue;
187 }
188 y = y1 - (points[j]/width);
189 distance = x*x + bias*y*y;
190 if (distance < shortest) {
191 next = j;
192 shortest = distance;
193 }
194 }
195 if (i%100 == 0) {
196 printf("%2.1f%%\n", (100.0*i)/(1.0*n_points));
197 }
198 /* swap points */
199 stor = points[next];
200 points[next] = points[i+1];
201 points[i+1] = stor;
202 }

301int
302hilbert_curve(EmbPattern *pattern, int iterations)
303{
304 char *state;
305 int i, position[2], direction;
306 EmbReal scale = 1.0;
307
308 /* Make the n-th iteration. */
309 state = malloc(MAX_STITCHES*10);
310 lindenmayer_system(hilbert_curve_l_system, state, iterations, 0);
311
312 /* Convert to an embroidery pattern. */
313 position[0] = 0;
314 position[1] = 0;
315 direction = 0;
316
317 for (i = 0; i < (int)strlen(state); i++) {
318 if (state[i] == '+') {
319 direction = (direction + 1) % 4;
320 continue;
321 }
322 if (state[i] == '-') {
323 direction = (direction + 3) % 4;
324 continue;
325 }
326 if (state[i] == 'F') {
327 int flags = NORMAL;
328 switch (direction) {
329 case 0:
330 default:
331 position[0]--;
332 break;
333 case 1:
334 position[1]++;
335 break;
336 case 2:
337 position[0]++;
338 break;
339 case 3:
340 position[1]--;
341 break;
342 }
343 embPattern_addStitchAbs(pattern, position[0]*scale, position[1]*scale, flags, 0);
344 }
345 }
346 free(state);
347 embPattern_end(pattern);
348 return 0;
349}

358{

366 for (i=length-1; i>=0; i--) {
367 state[2*i+1] = state[i];

369 state[2*i] = 'R';
370 } else {
371 state[2*i] = 'L';
372 }
373 }
374 state[2*length+1] = 0;
375 generate_dragon_curve(state, iterations-1);

632{

742EmbReal
743DistancePointLine(EmbVector p, EmbVector a, EmbVector b)
744{
745 /* if start == end, then use point-to-point distance */
746 if (a.X == b.X && a.Y == b.Y)
747 return DistancePointPoint(p, a);
748
749 /* otherwise use comp.graphics.algorithms Frequently Asked Questions method */
750 /* (1) AC dot AB
751 r = ---------
752 ||AB||^2
753
754 r has the following meaning:
755 r=0 Point = A
756 r=1 Point = B
757 r<0 Point is on the backward extension of AB
758 r>1 Point is on the forward extension of AB
759 0<r<1 Point is interior to AB
760 */
761
762 EmbReal r = ((p.X - a.X) * (b.X - a.X) + (p.Y - a.Y) * (b.Y - a.Y))
763 /
764 ((b.X - a.X) * (b.X - a.X) + (b.Y - a.Y) * (b.Y - a.Y));
765
766 if (r <= 0.0) return DistancePointPoint(p, a);
767 if (r >= 1.0) return DistancePointPoint(p, b);
768
769
770 /* (2)
771 (Ay-Cy)(Bx-Ax)-(Ax-Cx)(By-Ay)
772 s = -----------------------------
773 Curve^2
774
775 Then the distance from C to Point = |s|*Curve.
776 */
777
778 EmbReal s = ((a.Y - p.Y) * (b.X - a.X) - (a.X - p.X) * (b.Y - a.Y))
779 /
780 ((b.X - a.X) * (b.X - a.X) + (b.Y - a.Y) * (b.Y - a.Y));
781
782 return fabs(s) * sqrt(((b.X - a.X) * (b.X - a.X) + (b.Y - a.Y) * (b.Y - a.Y)));
783}

791public Vertices
792ReduceByArea(Vertices vertices, float areaTolerance)
793{
794 if (vertices.Count <= 3)
795 return vertices;
796
797 if (areaTolerance < 0)
798 {
799 throw new ArgumentOutOfRangeException("areaTolerance", "must be equal to or greater then zero.");
800 }
801
802 int result = new Vertices();
803 Vector2 v3;
804 Vector2 v1 = vertices[vertices.Count - 2];
805 Vector2 v2 = vertices[vertices.Count - 1];
806 areaTolerance *= 2;
807 for (int index = 0; index < vertices.Count; ++index, v2 = v3)
808 {
809 if (index == vertices.Count - 1)
810 {
811 if (result.Count == 0)
812 {
813 throw new ArgumentOutOfRangeException("areaTolerance", "The tolerance is too high!");
814 }
815 v3 = result[0];
816 }
817 else
818 {
819 v3 = vertices[index];
820 }
821 float old1, old2, new1;
822 MathUtils.Cross(ref v1, ref v2, out old1);
823 MathUtils.Cross(ref v2, ref v3, out old2);
824 MathUtils.Cross(ref v1, ref v3, out new1);
825 if (fabs(new1 - (old1 + old2)) > areaTolerance)
826 {
827 result.Add(v2);
828 v1 = v2;
829 }
830 }
831 return result;
832}

840void
841MergeParallelEdges(EmbArray *vertices, float tolerance)
842{
843 int i;
844 if (vertices.Count <= 3) {
845 /* Can't do anything useful here to a triangle. */
846 return;
847 }
848
849 int mergeMe = new bool[vertices.Count];
850 int newNVertices = vertices.Count;
851
852 /* Gather points to process */
853 for (i = 0; i < vertices->count; i++) {
854 EmbVector delta0, delta1;
855 int lower = (i == 0) ? (vertices.Count - 1) : (i - 1);
856 int upper = (i == vertices.Count - 1) ? (0) : (i + 1);
857
858 delta0 = embVector_subtract(vertices[i], vertices[lower]);
859 delta1 = embVector_subtract(vertices[upper], vertices[i]);
860 float norm0 = embVector_length(delta0);
861 float norm1 = embVector_length(delta0);
862
863 if (!(norm0 > 0.0f && norm1 > 0.0f) && newNVertices > 3) {
864 /* Merge identical points */
865 mergeMe[i] = 1;
866 newNVertices--;
867 }
868
869 delta0 = embVector_normalize(delta0);
870 delta1 = embVector_normalize(delta1);
871 float cross = embVector_cross(delta0, delta1);
872 float dot = embVector_dot(delta0, delta1);
873
874 if (fabs(cross) < tolerance && dot > 0 && newNVertices > 3) {
875 mergeMe[i] = 1;
876 newNVertices--;
877 }
878 else {
879 mergeMe[i] = 0;
880 }
881 }
882
883 if (newNVertices == vertices.Count || newNVertices == 0)
884 return;
885
886 int currIndex = 0;
887
888 /* Copy the vertices to a new list and clear the old */
889 int oldVertices = new Vertices(vertices);
890 vertices.Clear();
891
892 for (i = 0; i < oldVertices.Count; i++) {
893 if (mergeMe[i] || newNVertices == 0 || currIndex == newNVertices)
894 continue;
895
896 vertices.Add(oldVertices[i]);
897 currIndex++;
898 }
899}

914{
915 int i;

917 if (vertices->count < 3) {
918 embArray_copy(simplified, vertices);
919 return;
920 }
921
922 for (i = 0; i < vertices->count; i++) {
923 EmbVector delta;

925
926 delta = embVector_subtract(
927 vertices->geometry[nextId].object.vector,
928 vertices->geometry[i].object.vector);

932 continue;
933 }
