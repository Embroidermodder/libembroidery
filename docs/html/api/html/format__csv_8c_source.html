68int csvStrToStitchFlag(const char* str)
69{
70 if (!str) {
71 printf("ERROR: format-csv.c csvStrToStitchFlag(), str argument is null\n");
72 return -1;
73 }
74 if (!strcmp(str, "STITCH")) {
75 return NORMAL;
76 } else if (!strcmp(str, "JUMP")) {
77 return JUMP;
78 } else if (!strcmp(str, "TRIM")) {

82 } else if (!strcmp(str, "END")) {

85 return STOP;

90char
91readCsv(EmbPattern* pattern, FILE* file) {
92 int numColorChanges = 0;
93 int pos = 0;
94 int c = 0;
95 int cellNum = 0;
96 int process = 0;
97 int csvMode = CSV_MODE_NULL;
98 int expect = CSV_EXPECT_QUOTE1;
99 int flags = 0;
100 EmbReal xx = 0.0;
101 EmbReal yy = 0.0;
102 unsigned char r = 0, g = 0, b = 0;
103 char buff[100];
104
105 pos = 0;
106 while (!feof(file)) {
107 c = fgetc(file);
108 switch(c) {
109 case '"':
110 if (expect == CSV_EXPECT_QUOTE1) {
111 expect = CSV_EXPECT_QUOTE2;
112 } else if (expect == CSV_EXPECT_QUOTE2) {
113 expect = CSV_EXPECT_COMMA;
114 }
115 break;
116 case ',':
117 if (expect == CSV_EXPECT_COMMA) {
118 process = 1;
119 }
120 break;
121 case '\n':
122 if (expect == CSV_EXPECT_COMMA) {
123 process = 1;
124 } else if (expect == CSV_EXPECT_QUOTE1) {
125 /* Do Nothing. We encountered a blank line. */
126 } else {
127 printf("ERROR: format-csv.c readCsv(), premature newline\n");
128 return 0;
129 }
130 break;
131 default:
132 break;
133 }
134
135 if (process) {
136 buff[pos] = 0;
137 pos = 0;
138 process = 0;
139 cellNum++;
140 expect = CSV_EXPECT_QUOTE1;
141 if (csvMode == CSV_MODE_NULL) {
142 if (!strcmp(buff, "#")) {
143 csvMode = CSV_MODE_COMMENT;
144 } else if (!strcmp(buff, ">")) {
145 csvMode = CSV_MODE_VARIABLE;
146 } else if (!strcmp(buff, "$")) {
147 csvMode = CSV_MODE_THREAD;
148 } else if (!strcmp(buff, "*")) {
149 csvMode = CSV_MODE_STITCH;
150 } else {/* TODO: error */
151 return 0;
152 }
153 } else if (csvMode == CSV_MODE_COMMENT) {
154 /* Do Nothing */
155 } else if (csvMode == CSV_MODE_VARIABLE) {
156 /* Do Nothing */
157 } else if (csvMode == CSV_MODE_THREAD) {
158 if (cellNum == 2) {
159 /* Do Nothing. Ignore Thread Number */
160 } else if (cellNum == 3) {
161 r = (unsigned char)atoi(buff);
162 } else if (cellNum == 4) {
163 g = (unsigned char)atoi(buff);
164 } else if (cellNum == 5) {
165 b = (unsigned char)atoi(buff);
166 } else if (cellNum == 6) {
167 /* TODO: Thread Description */
168 } else if (cellNum == 7) {
169 /* TODO: Thread Catalog Number */
170 EmbThread t;
171 t.color.r = r;
172 t.color.g = g;
173 t.color.b = b;
174 strcpy(t.description, "TODO:DESCRIPTION");
175 strcpy(t.catalogNumber, "TODO:CATALOG_NUMBER");
176 embPattern_addThread(pattern, t);
177 csvMode = CSV_MODE_NULL;
178 cellNum = 0;
179 } else {
180 /* TODO: error */
181 return 0;
182 }
183 } else if (csvMode == CSV_MODE_STITCH) {
184 /* printf("%s\n", buff); */
185 if (cellNum == 2) {
186 flags = csvStrToStitchFlag(buff);
187 if (flags == STOP) {
188 numColorChanges++;
189 }
190 } else if (cellNum == 3) {
191 xx = atof(buff);
192 } else if (cellNum == 4) {
193 yy = atof(buff);
194 /* printf("%d %d %f %f\n",
195 pattern->stitch_list->count, flags, xx, yy); */
196 embPattern_addStitchAbs(pattern, xx, yy, flags, 1);
197 csvMode = CSV_MODE_NULL;
198 cellNum = 0;
199 } else {
200 /* TODO: error */
201 return 0;
202 }
203 }
204 if (c == '\n') {
205 csvMode = CSV_MODE_NULL;
206 cellNum = 0;
207 }
208 } else {
209 if (expect == CSV_EXPECT_QUOTE2 && c != '"') {
210 buff[pos++] = (char)c;
211 }
212 }
213 }
214
215 return 1;
216}

225#if LIBEMBROIDERY_EMBEDDED_VERSION

239 fprintf(file, "\"#\",\"General Notes:\"\n");
240 fprintf(file, "\"#\",\"This file can be read by Excel or LibreOffice as CSV (Comma Separated Value) or with a text editor.\"\n");
241 fprintf(file, "\"#\",\"Lines beginning with # are comments.\"\n");
242 fprintf(file, "\"#\",\"Lines beginning with > are variables: [VAR_NAME], [VAR_VALUE]\"\n");
243 fprintf(file, "\"#\",\"Lines beginning with $ are threads: [THREAD_NUMBER], [RED], [GREEN], [BLUE], [DESCRIPTION], [CATALOG_NUMBER]\"\n");
244 fprintf(file, "\"#\",\"Lines beginning with * are stitch entries: [STITCH_TYPE], [X], [Y]\"\n");
245 fprintf(file, "\"#\",\" \"\n");
246 fprintf(file, "\"#\",\"Stitch Entry Notes:\"\n");
247 fprintf(file, "\"#\",\"STITCH instructs the machine to move to the position [X][Y] and then make a stitch.\"\n");
248 fprintf(file, "\"#\",\"JUMP instructs the machine to move to the position [X][Y] without making a stitch.\"\n");
249 fprintf(file, "\"#\",\"TRIM instructs the machine to cut the thread before moving to the position [X][Y] without making a stitch.\"\n");
250 fprintf(file, "\"#\",\"COLOR instructs the machine to stop temporarily so that the user can change to a different color thread before resuming.\"\n");
251 fprintf(file, "\"#\",\"END instructs the machine that the design is completed and there are no further instructions.\"\n");
252 fprintf(file, "\"#\",\"UNKNOWN encompasses instructions that may not be supported currently.\"\n");
253 fprintf(file, "\"#\",\"[X] and [Y] are absolute coordinates in millimeters (mm).\"\n");
254 fprintf(file, "\"#\",\" \"\n");
255#endif

258 fprintf(file,"\"#\",\"[VAR_NAME]\",\"[VAR_VALUE]\"\n");
259 fprintf(file, "\">\",\"STITCH_COUNT:\",\"%u\"\n", (unsigned int)pattern->stitch_list->count);
260 fprintf(file, "\">\",\"THREAD_COUNT:\",\"%u\"\n", (unsigned int)pattern->thread_list->count);

266 fprintf(file, "\">\",\"EXTENTS_HEIGHT:\",\"%f\"\n", boundingRect.bottom - boundingRect.top);
267 fprintf(file,"\n");
268
269 /* write colors */

271 fprintf(file, "\"[BLUE]\",\"[DESCRIPTION]\",\"[CATALOG_NUMBER]\"\n");
272
273 for (i = 0; i < pattern->thread_list->count; i++) {
274 EmbThread thr = pattern->thread_list->thread[i];
275 /* TODO: fix segfault that backtraces here when

281 (int)thr.color.b,
282 thr.description,
283 thr.catalogNumber);
284 }
285 fprintf(file, "\n");
286
287 /* write stitches */

289 for (i = 0; i < pattern->stitch_list->count; i++) {
290 EmbStitch s = pattern->stitch_list->stitch[i];
291 fprintf(file, "\"*\",\"%s\",\"%f\",\"%f\"\n",
