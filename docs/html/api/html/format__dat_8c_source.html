22char
23readDat(EmbPattern* pattern, FILE* file)
24{
25 unsigned char b0;
26 int fileLength, stitchesRemaining, b1, b2, stitchType;
27
28 fseek(file, 0x00, SEEK_END);
29 fileLength = ftell(file);
30 if (fileLength < 0x100) {
31 puts("ERROR: dat file too short to contain header.");
32 return 0;
33 }
34 fseek(file, 0x02, SEEK_SET);
35 stitchesRemaining = fread_uint16(file);
36 if (emb_verbose>1) {
37 printf("stitchesRemaining: %d", stitchesRemaining);
38 }
39 fseek(file, 0x100, SEEK_SET);
40
41 while (!feof(file)) {
42 b1 = (int)fgetc(file);
43 b2 = (int)fgetc(file);
44 b0 = fgetc(file);
45
46 stitchType = NORMAL;
47
48 if ((b0 & 0x02) == 0) stitchType = TRIM;
49
50 if (b0 == 0x87) {
51 stitchType = STOP;
52 }
53 if (b0 == 0xF8) {
54 break;
55 }
56 if (b1 >= 0x80) {
57 b1 = -(b1 & 0x7F);
58 }
59 if (b2 >= 0x80) {
60 b2 = -(b2 & 0x7F);
61 }
62 embPattern_addStitchRel(pattern, b1 / 10.0, b2 / 10.0, stitchType, 1);
63 }
64
65 return 1;
66}
