148void
149encode_record(FILE* file, int x, int y, int flags)
150{
151 unsigned char b[3];
152 encode_tajima_ternary(b, x, y);
153
154 b[2] |= 0x03;
155
156 if (flags & END) {
157 /* 0b10001101 = -13 = 0x8D */
158 b[0] = 0;
159 b[1] = 0;
160 b[2] = 0x8D;
161 }
162
163 /* make the variable v configurable for the pattern */
164 if (flags & TRIM) {
165 int dx, dy, i, v = 5;
166 dx = x/v;
167 dy = y/v;
168 for (i = 1; i < v; i++) {
169 encode_record(file, dx, dy, JUMP);
170 }
171 encode_record(file, x - (dx * (v - 1)), y - (dy * (v - 1)), JUMP);
172 return;
173 }
174 if (flags & (JUMP | TRIM)) {
175 b[2] = (char) (b[2] | 0x83);
176 }
177 if (flags & STOP) {
178 b[2] = (char) (b[2] | 0xC3);
179 }
180
181 fwrite(b, 1, 3, file);
182}

251/*
252 * The header seems to contain information about the design.
253 * Seems to be ASCII text delimited by 0x0D (carriage returns).
254 * This must be in the file for most new software or hardware
255 * to consider it a good file! This is much more important
256 * than I originally believed. The header is 125 bytes in
257 * length and padded out by 0x20 to 512 bytes total.
258 * All entries in the header seem to be 2 ASCII characters
259 * followed by a colon, then it's value trailed by a carriage return.
260 *
261 * char LA[16+1]; First is the 'LA' entry, which is the design name with no
262 * path or extension information. The blank is 16 characters
263 * in total, but the name must not be longer that 8 characters
264 * and padded out with 0x20.
265 *
266 * char ST[7+1]; Next is the stitch count ST, this is a 7 digit number
267 * padded by leading zeros. This is the total stitch count
268 * including color changes, jumps, nups, and special records.
269 *
270 * char CO[3+1]; Next, is CO or colors, a 3 digit number padded by leading
271 * zeros. This is the number of color change records in the file.
272 *
273 * char POSX[5+1]; Next is +X or the positive X extent in centimeters, a 5
274 * digit non-decimal number padded by leading zeros.
275 *
276 * char NEGX[5+1]; Following is the -X or the negative X extent in millimeters,
277 * a 5 digit non-decimal number padded by leading zeros.
278 *
279 * char POSY[5+1]; Again, the +Y extents.
280 *
281 * char NEGY[5+1]; Again, the -Y extents.
282 *
283 * char AX[6+1]; AX and AY should express the relative coordinates of the
284 * char AY[6+1]; last point from the start point in 0.1 mm. If the start
285 * and last points are the same, the coordinates are (0,0).
286 *
287 * char MX[6+1]; MX and MY should express coordinates of the last point of
288 * char MY[6+1]; the previous file for a multi-volume design. A multi-
289 * volume design means a design consisted of two or more files.
290 * This was used for huge designs that can not be stored in a
291 * single paper tape roll. It is not used so much (almost
292 * never) nowadays.
293 *
294 * char PD[9+1]; PD is also storing some information for multi-volume design.
295 */

297readDst(EmbPattern* pattern, FILE* file) {
298 char var[3]; /* temporary storage variable name */
299 char val[512]; /* temporary storage variable value */
300 int valpos;
301 unsigned char b[3];
302 char header[512 + 1];
303 int i = 0, flags;
304
305 /* TODO: review commented code below
306 pattern->clear();
307 pattern->set_variable("file_name",filename);
308 */
309
310 fread(header, 1, 512, file);
311
312 /*TODO:It would probably be a good idea to
313 validate file before accepting it. */
314 /* fill variables from header fields */
315 for (i = 0; i < 512; i++) {
316 if (header[i] == ':' && i > 1) {
317 var[0] = header[i - 2];
318 var[1] = header[i - 1];
319 var[2] = '\0';
320 valpos = i + 1;
321 for (i++; i < 512; i++)
322 {
323 /* don't accept : without CR because there's a bug below: i-valpos must be > 0 which is not the case if the : is before the third character. */
324 if (header[i] == 13/*||header[i]==':'*/) { /* 0x0d = carriage return */
325 /* : indicates another variable, CR was missing! */
326 if (header[i] == ':') {
327 i -= 2;
328 }
329 strncpy(val, &header[valpos], (size_t)(i - valpos));
330 val[i - valpos] = '\0';
331 set_dst_variable(pattern, var, val);
332 break;
333 }
334 }
335 }
336 }
337
338 while (fread(b, 1, 3, file) == 3) {
339 int x, y;
340 decode_tajima_ternary(b, &x, &y);
341 flags = decode_record_flags(b[2]);
342 if (flags == END) {
343 break;
344 }
345 embPattern_addStitchRel(pattern, x / 10.0, y / 10.0, flags, 1);
346 }
347
348 /* combineJumpStitches(pattern, 5); */
349 return 1;
350}
