25char
26readDsz(EmbPattern* pattern, FILE* file)
27{
28 unsigned char b[3];
29
30 fseek(file, 0x200, SEEK_SET);
31 while (fread(b, 1, 3, file) == 3) {
32 int x, y;
33 unsigned char ctrl;
34 int stitchType = NORMAL;
35
36 y = b[0];
37 x = b[1];
38 ctrl = b[2];
39 if (ctrl & 0x01) {
40 stitchType = TRIM;
41 }
42 if (ctrl & 0x20) {
43 y = -y;
44 }
45 if (ctrl & 0x40) {
46 x = -x;
47 }
48 if (ctrl & 0x0E) {
49 /*
50 TODO: Add back if head number is added to the library
51 int headNumber = (ctrl & 0x0E) >> 1;
52 */
53 stitchType = STOP;
54 }
55 if (ctrl & 0x10) {
56 embPattern_addStitchRel(pattern, 0, 0, END, 1);
57 break;
58 }
59 embPattern_addStitchRel(pattern, x / 10.0, y / 10.0, stitchType, 1);
60 }
61 return 1;
62}

67char
68writeDsz(EmbPattern* pattern, FILE* file)
69{
70 int i;
71 EmbVector delta;
72 EmbVector start;
73 fpad(file, ' ', 200);
74 start.x = 0.0;
75 start.y = 0.0;
76 for (i=0; i<pattern->stitch_list->count; i++) {
77 EmbVector pos;
78 EmbStitch st;
79 unsigned char b[3];
80 st = pattern->stitch_list->stitch[i];
81 pos.x = st.x;
82 pos.y = st.y;
83 delta = embVector_subtract(pos, start);
84 b[2] = 0;
85 if (st.flags & TRIM) {
86 b[2] |= 0x01;
87 }
88 if (10.0*delta.x < 0.0) {
89 b[2] |= 0x20;
90 }
91 if (10.0*delta.y < 0.0) {
92 b[2] |= 0x40;
93 }
94 if (st.flags & STOP) {
95 b[2] |= 0x0E;
96 }
97 if (st.flags & END) {
98 b[2] |= 0x10;
99 }
100 b[0] = emb_round(10.0*delta.x);
101 b[1] = emb_round(10.0*delta.y);
102 fwrite(b, 1, 3, file);
103 }
104 return 1;
105}
