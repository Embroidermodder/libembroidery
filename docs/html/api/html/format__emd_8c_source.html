34 unsigned char jemd0[6]; /* TODO: more descriptive name */

51
52 while (!feof(file)) {
53 char dx, dy;
54 int flags = NORMAL;
55 fread(b, 1, 2, file);
56
57 if (b[0] == 0x80) {
58 if (b[1] == 0x2A) {
59 embPattern_addStitchRel(pattern, 0, 0, STOP, 1);
60 continue;
61 } else if (b[1] == 0x80) {
62 fread(b, 1, 2, file);
63 flags = TRIM;
64 } else if (b[1] == 0xFD) {
65 embPattern_addStitchRel(pattern, 0, 0, END, 1);
66 break;
67 } else {
68 continue;
69 }
70 }
71 dx = emdDecode(b[0]);
72 dy = emdDecode(b[1]);
73 embPattern_addStitchRel(pattern, dx / 10.0, dy / 10.0, flags, 1);
74 }
75 return 1;
76}
