24int decode_exy_flags(unsigned char b2)
25{
26 int returnCode = 0;
27 if (b2 == 0xF3) {
28 return (END);
29 }
30 if ((b2 & 0xC3) == 0xC3) {
31 return TRIM | STOP;
32 }
33 if (b2 & 0x80) {
34 returnCode |= TRIM;
35 }
36 if (b2 & 0x40) {
37 returnCode |= STOP;
38 }
39 return returnCode;
40}
