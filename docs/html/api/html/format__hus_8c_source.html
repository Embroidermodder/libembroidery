28int husDecodeStitchType(unsigned char b)
29{
30 switch(b) {
31 case 0x80:
32 return NORMAL;
33 case 0x81:
34 return JUMP;
35 case 0x84:
36 return STOP;
37 case 0x90:
38 return END;
39 default:
40 return NORMAL;
41 }
42}

44unsigned char*
45husDecompressData(unsigned char* input, int compressedInputLength, int decompressedContentLength)
46{
47 char* decompressedData = (char*)malloc(sizeof(char)*decompressedContentLength);
48 if (!decompressedData) { printf("ERROR: husDecompressData(), cannot allocate memory for decompressedData\n"); return 0; }
49 hus_decompress((char*)input, compressedInputLength, decompressedData, &decompressedContentLength);
50 return (unsigned char *)decompressedData;

93readHus(EmbPattern* pattern, FILE* file)
94{
95 int fileLength;
96 int magicCode, numberOfStitches, numberOfColors;
97 int positiveXHoopSize, positiveYHoopSize, negativeXHoopSize, negativeYHoopSize;
98 int attributeOffset, xOffset, yOffset;
99 unsigned char *attributeData, *xData , *yData;
100 unsigned char* attributeDataDecompressed = 0;
101 unsigned char* xDecompressed = 0;
102 unsigned char* yDecompressed = 0;
103 unsigned char* stringVal = 0;
104
105 int unknown, i = 0;
106
107 puts("ERROR: Overridden. readHus is not implemented for now.");
108 return 0;
109
110 fseek(file, 0x00, SEEK_END);
111 fileLength = ftell(file);
112 fseek(file, 0x00, SEEK_SET);
113
114 embInt_read(file, "magicCode", &magicCode, EMB_INT32_LITTLE);
115 embInt_read(file, "numberOfStitchs", &numberOfStitches, EMB_INT32_LITTLE);
116 embInt_read(file, "numberOfColors", &numberOfColors, EMB_INT32_LITTLE);
117
118 /* These should be put in the EmbPattern. */
119 embInt_read(file, "+X HoopSize", &positiveXHoopSize, EMB_INT16_LITTLE);
120 embInt_read(file, "+Y HoopSize", &positiveYHoopSize, EMB_INT16_LITTLE);
121 embInt_read(file, "-X HoopSize", &negativeXHoopSize, EMB_INT16_LITTLE);
122 embInt_read(file, "-Y HoopSize", &negativeYHoopSize, EMB_INT16_LITTLE);
123
124 embInt_read(file, "attributeOffset", &attributeOffset, EMB_INT32_LITTLE);
125 embInt_read(file, "xOffset", &xOffset, EMB_INT32_LITTLE);
126 embInt_read(file, "yOffset", &yOffset, EMB_INT32_LITTLE);
127
128 stringVal = (unsigned char*)malloc(sizeof(unsigned char)*8);
129 if (!stringVal) {
130 printf("ERROR: format-hus.c readHus(), cannot allocate memory for stringVal\n");
131 return 0;
132 }
133 fread(stringVal, 1, 8, file); /* TODO: check return value */
134
135 embInt_read(file, "unknown", &unknown, EMB_INT16_LITTLE);
136 for (i = 0; i < numberOfColors; i++) {
137 short pos;
138 embInt_read(file, "pos", &pos, EMB_INT16_LITTLE);
139 embPattern_addThread(pattern, husThreads[pos]);
140 }
141
142 attributeData = (unsigned char*)malloc(sizeof(unsigned char)*(xOffset - attributeOffset + 1));
143 if (!attributeData) {
144 printf("ERROR: format-hus.c readHus(), cannot allocate memory for attributeData\n");
145 return 0;
146 }
147 fread(attributeData, 1, xOffset - attributeOffset, file); /* TODO: check return value */
148 attributeDataDecompressed = husDecompressData(attributeData, xOffset - attributeOffset, numberOfStitches + 1);
149
150 xData = (unsigned char*)malloc(sizeof(unsigned char)*(yOffset - xOffset + 1));
151 if (!xData) {
152 printf("ERROR: format-hus.c readHus(), cannot allocate memory for xData\n");
153 return 0;
154 }
155 fread(xData, 1, yOffset - xOffset, file); /* TODO: check return value */
156 xDecompressed = husDecompressData(xData, yOffset - xOffset, numberOfStitches);
157
158 yData = (unsigned char*)malloc(sizeof(unsigned char)*(fileLength - yOffset + 1));
159 if (!yData) {
160 printf("ERROR: format-hus.c readHus(), cannot allocate memory for yData\n");
161 return 0;
162 }
163 fread(yData, 1, fileLength - yOffset, file); /* TODO: check return value */
164 yDecompressed = husDecompressData(yData, fileLength - yOffset, numberOfStitches);
165
166 for (i = 0; i < numberOfStitches; i++) {
167 int flag;
168 EmbVector v;
169 v.x = husDecodeByte(xDecompressed[i]) / 10.0;
170 v.y = husDecodeByte(yDecompressed[i]) / 10.0;
171 flag = husDecodeStitchType(attributeDataDecompressed[i]);
172 embPattern_addStitchRel(pattern, v.x, v.y, flag, 1);
173 }
174
175 safe_free(stringVal);
176 safe_free(xData);
177 safe_free(xDecompressed);
178 safe_free(yData);
179 safe_free(yDecompressed);
180 safe_free(attributeData);
181 safe_free(attributeDataDecompressed);
182
183 return 1;
184}

186char
187writeHus(EmbPattern* pattern, FILE* file)
188{
189 EmbRect boundingRect;
190 int stitchCount, minColors, patternColor, attributeSize, xCompressedSize, yCompressedSize, i;
191 EmbReal previousX, previousY;
192 short right, top, bottom, left;
193 unsigned int code, colors, offset1, offset2;
194 unsigned char *xValues = 0, *yValues = 0, *attributeValues = 0,
195 *attributeCompressed = 0, *xCompressed = 0, *yCompressed = 0;
196
197 stitchCount = pattern->stitch_list->count;
198 /* embPattern_correctForMaxStitchLength(pattern, 0x7F, 0x7F); */
199 minColors = pattern->thread_list->count;
200 patternColor = minColors;
201 if (minColors > 24) minColors = 24;
202 code = 0x00C8AF5B;
203 embInt_write(file, "code", &code, EMB_INT32_LITTLE);
204 embInt_write(file, "stitchCount", &stitchCount, EMB_INT32_LITTLE);
205 embInt_write(file, "minColors", &minColors, EMB_INT32_LITTLE);
206
207 boundingRect = embPattern_calcBoundingBox(pattern);
208 right = (short) emb_round(boundingRect.right * 10.0);
209 top = (short) -emb_round(boundingRect.top * 10.0);
210 left = (short) emb_round(boundingRect.left * 10.0);
211 bottom = (short) -emb_round(boundingRect.bottom * 10.0);
212 embInt_write(file, "right", &right, EMB_INT16_LITTLE);
213 embInt_write(file, "top", &top, EMB_INT16_LITTLE);
214 embInt_write(file, "left", &left, EMB_INT16_LITTLE);
215 embInt_write(file, "bottom", &bottom, EMB_INT16_LITTLE);
216
217 colors = 0x2A + 2 * minColors;
218 embInt_write(file, ".", &colors, EMB_INT32_LITTLE);
219
220 xValues = (unsigned char*)malloc(sizeof(unsigned char)*(stitchCount));
221 if (!xValues) {
222 printf("ERROR: format-hus.c writeHus(), cannot allocate memory for xValues\n");
223 return 0;
224 }
225 yValues = (unsigned char*)malloc(sizeof(unsigned char)*(stitchCount));
226 if (!yValues) {
227 printf("ERROR: format-hus.c writeHus(), cannot allocate memory for yValues\n");
228 return 0;
229 }
230 attributeValues = (unsigned char*)malloc(sizeof(unsigned char)*(stitchCount));
231 if (!attributeValues) {
232 printf("ERROR: format-hus.c writeHus(), cannot allocate memory for attributeValues\n");
233 return 0;
234 }
235
236 previousX = 0.0;
237 previousY = 0.0;
238 for (i = 0; i < pattern->stitch_list->count; i++) {
239 EmbStitch st;
240 st = pattern->stitch_list->stitch[i];
241 xValues[i] = husEncodeByte((st.x - previousX) * 10.0);
242 previousX = st.x;
243 yValues[i] = husEncodeByte((st.y - previousY) * 10.0);
244 previousY = st.y;
245 attributeValues[i] = husEncodeStitchType(st.flags);
246 }
247 attributeCompressed = husCompressData(attributeValues, stitchCount, &attributeSize);
248 xCompressed = husCompressData(xValues, stitchCount, &xCompressedSize);
249 yCompressed = husCompressData(yValues, stitchCount, &yCompressedSize);
250 /* TODO: error if husCompressData returns zero? */
251
252 offset1 = (unsigned int) (0x2A + 2 * patternColor + attributeSize);
253 offset2 = (unsigned int) (0x2A + 2 * patternColor + attributeSize + xCompressedSize);
254 embInt_write(file, "offset1", &offset1, EMB_INT32_LITTLE);
255 embInt_write(file, "offset2", &offset2, EMB_INT32_LITTLE);
256 fpad(file, 0, 10);
257
258 for (i = 0; i < patternColor; i++) {
259 short color_index = (short)embThread_findNearestThread(pattern->thread_list->thread[i].color, (EmbThread*)husThreads, 29);
260 embInt_write(file, ".", &color_index, EMB_INT16_LITTLE);
261 }
262
263 fwrite(attributeCompressed, 1, attributeSize, file);
264 fwrite(xCompressed, 1, xCompressedSize, file);
265 fwrite(yCompressed, 1, yCompressedSize, file);
266
267 safe_free(xValues);
268 safe_free(xCompressed);
269 safe_free(yValues);
270 safe_free(yCompressed);
271 safe_free(attributeValues);
272 safe_free(attributeCompressed);
273
274 return 1;
275}
