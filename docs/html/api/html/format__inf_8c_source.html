27char
28readInf(EmbPattern* pattern, FILE* file)
29{
30 int nColors, i;
31 char colorType[50];
32 char colorDescription[50];
33 EmbThread t;
34
35 fseek(file, 12, SEEK_CUR);
36 embInt_read(file, "nColors", &nColors, EMB_INT32_BIG);
37
38 pattern->thread_list->count = 0;
39
40 for (i = 0; i < nColors; i++) {
41 fseek(file, 4, SEEK_CUR);
42 embColor_read(file, &(t.color), 3);
43 strcpy(t.catalogNumber, "");
44 strcpy(t.description, "");
45 embPattern_addThread(pattern, t);
46 fseek(file, 2, SEEK_CUR);
47 binaryReadString(file, colorType, 50);
48 binaryReadString(file, colorDescription, 50);
49 }
50 return 1;
51}

53char
54writeInf(EmbPattern* pattern, FILE* file)
55{
56 int i, bytesRemaining;
57
58 binaryWriteUIntBE(file, 0x01);
59 binaryWriteUIntBE(file, 0x08);
60 /* write place holder offset */
61 binaryWriteUIntBE(file, 0x00);
62 binaryWriteUIntBE(file, pattern->thread_list->count);
63
64 for (i = 0; i < pattern->thread_list->count; i++) {
65 char buffer[50];
66 unsigned short record_length, record_number, needle_number;
67 EmbColor c;
68 c = pattern->thread_list->thread[i].color;
69 sprintf(buffer, "RGB(%d,%d,%d)", (int)c.r, (int)c.g, (int)c.b);
70 record_length = 14 + strlen(buffer);
71 record_number = i;
72 needle_number = i;
73 embInt_write(file, "record length", &record_length, EMB_INT16_BIG);
74 embInt_write(file, "record number", &record_number, EMB_INT16_BIG);
75 embColor_write(file, c, 3);
76 embInt_write(file, "needle number", &needle_number, EMB_INT16_BIG);
77 fwrite("RGB\0", 1, 4, file);
78 fprintf(file, "%s", buffer);
79 fwrite("\0", 1, 1, file);
80 }
81 /* It appears that there should be a pad here otherwise it clips into
82 * the color description. */
83 fpad(file, 0, 8);
84 fseek(file, -8, SEEK_END);
85 bytesRemaining = ftell(file);
86 fseek(file, 8, SEEK_SET);
87 binaryWriteUIntBE(file, bytesRemaining);
88 return 1;
89}
