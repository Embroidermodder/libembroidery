23int
24jefGetHoopSize(int width, int height)
25{
26 if (width < 50 && height < 50) {
27 return HOOP_50X50;
28 }
29 if (width < 110 && height < 110) {
30 return HOOP_110X110;
31 }
32 if (width < 140 && height < 200) {
33 return HOOP_140X200;
34 }
35 return ((int) HOOP_110X110);
36}

47jefSetHoopFromId(EmbPattern* pattern, int hoopCode)
48{
49 if (!pattern) {
50 printf("ERROR: format-jef.c jefSetHoopFromId(), pattern argument is null\n");
51 return;
52 }
53 switch (hoopCode) {
54 case HOOP_126X110:
55 pattern->hoop_height = 126.0;
56 pattern->hoop_width = 110.0;
57 break;
58 case HOOP_50X50:
59 pattern->hoop_height = 50.0;
60 pattern->hoop_width = 50.0;
61 break;
62 case HOOP_110X110:

64 pattern->hoop_width = 110.0;
65 break;

67 pattern->hoop_height = 140.0;
68 pattern->hoop_width = 200.0;
69 break;
70 case HOOP_230X200:
71 pattern->hoop_height = 230.0;
72 pattern->hoop_width = 200.0;

74 }

97char
98readJef(EmbPattern* pattern, FILE* file) {
99 int stitchOffset, formatFlags, numberOfColors, numberOfStitchs;
100 int hoopSize, i, stitchCount;
101 struct hoop_padding bounds, rectFrom110x110;
102 struct hoop_padding rectFrom50x50, rectFrom200x140, rect_from_custom;
103 char date[8], time[8];
104
105 embInt_read(file, "stitchOffset", &stitchOffset, EMB_INT32_LITTLE);
106 embInt_read(file, "formatFlags", &formatFlags, EMB_INT32_LITTLE); /* TODO: find out what this means */
107
108 fread((unsigned char*) date, 1, 8, file); /* TODO: check return value */
109 fread((unsigned char*) time, 1, 8, file); /* TODO: check return value */
110 embInt_read(file, "numberOfColors", &numberOfColors, EMB_INT32_LITTLE);
111 embInt_read(file, "numberOfStitchs", &numberOfStitchs, EMB_INT32_LITTLE);
112 embInt_read(file, "hoopSize", &hoopSize, EMB_INT32_LITTLE);
113 jefSetHoopFromId(pattern, hoopSize);
114 if (numberOfStitchs > MAX_STITCHES) {
115 numberOfStitchs = MAX_STITCHES;
116 puts("ERROR: this file is corrupted or has too many stitches.");
117 return 0;
118 }
119 if (emb_verbose>1) {
120 printf("format flags = %d\n", formatFlags);
121 printf("number of colors = %d\n", numberOfColors);
122 printf("number of stitches = %d\n", numberOfStitchs);
123 }
124
125 read_hoop(file, &bounds, "bounds");
126 read_hoop(file, &rectFrom110x110, "rectFrom110x110");
127 read_hoop(file, &rectFrom50x50, "rectFrom50x50");
128 read_hoop(file, &rectFrom200x140, "rectFrom200x140");
129 read_hoop(file, &rect_from_custom, "rect_from_custom");
130
131 for (i = 0; i < numberOfColors; i++) {
132 int thread_num;
133 embInt_read(file, "thread_num", &thread_num, EMB_INT32_LITTLE);
134 embPattern_addThread(pattern, jefThreads[thread_num % 79]);
135 }
136 fseek(file, stitchOffset, SEEK_SET);
137 stitchCount = 0;
138 while (stitchCount < numberOfStitchs + 100) {
139 unsigned char b[2];
140 char dx = 0, dy = 0;
141 int flags = NORMAL;
142 if (fread(b, 1, 2, file) != 2) {
143 break;
144 }
145
146 if (b[0] == 0x80) {
147 if (b[1] & 1) {
148 if (fread(b, 1, 2, file) != 2) {
149 break;
150 }
151 flags = STOP;
152 }
153 else if ((b[1] == 2) || (b[1] == 4) || b[1] == 6) {
154 if (fread(b, 1, 2, file) != 2) {
155 break;
156 }
157 flags = TRIM;
158 }
159 else if (b[1] == 0x10) {
160 embPattern_addStitchRel(pattern, 0.0, 0.0, END, 1);
161 break;
162 }
163 }
164 dx = jefDecode(b[0]);
165 dy = jefDecode(b[1]);
166 embPattern_addStitchRel(pattern, dx / 10.0, dy / 10.0, flags, 1);
167 stitchCount++;
168 }
169 return 1;
170}

172void
173jefEncode(unsigned char* b, char dx, char dy, int flags) {
174 if (!b) {
175 printf("ERROR: format-jef.c expEncode(), b argument is null\n");
176 return;
177 }
178 if (flags == STOP) {
179 b[0] = 0x80;
180 b[1] = 1;
181 b[2] = dx;
182 b[3] = dy;
183 } else if (flags == END) {
184 b[0] = 0x80;
185 b[1] = 0x10;
186 b[2] = 0;
187 b[3] = 0;
188 } else if (flags == TRIM || flags == JUMP) {
189 b[0] = 0x80;
190 b[1] = 2;
191 b[2] = dx;
192 b[3] = dy;
193 } else {
194 b[0] = dx;
195 b[1] = dy;
196 }
197}

199char
200writeJef(EmbPattern* pattern, FILE* file)
201{
202 int colorlistSize, minColors, designWidth, designHeight, i;
203 EmbRect boundingRect;
204 EmbTime time;
205 EmbReal width, height;
206 int data;
207 EmbVector pos;
208
209 embPattern_correctForMaxStitchLength(pattern, 12.7, 12.7);
210
211 colorlistSize = pattern->thread_list->count;
212 minColors = EMB_MAX(colorlistSize, 6);
213 binaryWriteInt(file, 0x74 + (minColors * 4));
214 binaryWriteInt(file, 0x0A);
215
216 embTime_initNow(&time);
217
218 fprintf(file, "%04d%02d%02d%02d%02d%02d", (int)(time.year + 1900),
219 (int)(time.month + 1), (int)(time.day), (int)(time.hour),
220 (int)(time.minute), (int)(time.second));
221 fpad(file, 0, 2);
222 embInt_write(file, ".", &(pattern->thread_list->count), EMB_INT32_LITTLE);
223 data = pattern->stitch_list->count + EMB_MAX(0, (6 - colorlistSize) * 2) + 1;
224 embInt_write(file, ".", &data, EMB_INT32_LITTLE);
225
226 boundingRect = embPattern_calcBoundingBox(pattern);
227 width = boundingRect.right - boundingRect.left;
228 height = boundingRect.bottom - boundingRect.top;
229 designWidth = (int)(width * 10.0);
230 designHeight = (int)(height * 10.0);
231
232 binaryWriteInt(file, jefGetHoopSize(designWidth, designHeight));
233
234 /* Distance from center of Hoop */
235 binaryWriteInt(file, (int) (designWidth / 2)); /* left */
236 binaryWriteInt(file, (int) (designHeight / 2)); /* top */
237 binaryWriteInt(file, (int) (designWidth / 2)); /* right */
238 binaryWriteInt(file, (int) (designHeight / 2)); /* bottom */
239
240 /* Distance from default 110 x 110 Hoop */
241 if (EMB_MIN(550 - designWidth / 2, 550 - designHeight / 2) >= 0) {
242 binaryWriteInt(file, EMB_MAX(-1, 550 - designWidth / 2)); /* left */
243 binaryWriteInt(file, EMB_MAX(-1, 550 - designHeight / 2)); /* top */
244 binaryWriteInt(file, EMB_MAX(-1, 550 - designWidth / 2)); /* right */
245 binaryWriteInt(file, EMB_MAX(-1, 550 - designHeight / 2)); /* bottom */
246 } else {
247 binaryWriteInt(file, -1);
248 binaryWriteInt(file, -1);
249 binaryWriteInt(file, -1);
250 binaryWriteInt(file, -1);
251 }
252
253 /* Distance from default 50 x 50 Hoop */
254 if (EMB_MIN(250 - designWidth / 2, 250 - designHeight / 2) >= 0) {
255 binaryWriteInt(file, (int) EMB_MAX(-1, 250 - designWidth / 2)); /* left */
256 binaryWriteInt(file, (int) EMB_MAX(-1, 250 - designHeight / 2)); /* top */
257 binaryWriteInt(file, (int) EMB_MAX(-1, 250 - designWidth / 2)); /* right */
258 binaryWriteInt(file, (int) EMB_MAX(-1, 250 - designHeight / 2)); /* bottom */
259 } else {
260 binaryWriteInt(file, -1);
261 binaryWriteInt(file, -1);
262 binaryWriteInt(file, -1);
263 binaryWriteInt(file, -1);
264 }
265
266 /* Distance from default 140 x 200 Hoop */
267 binaryWriteInt(file, (int) (700 - designWidth / 2)); /* left */
268 binaryWriteInt(file, (int) (1000 - designHeight / 2)); /* top */
269 binaryWriteInt(file, (int) (700 - designWidth / 2)); /* right */
270 binaryWriteInt(file, (int) (1000 - designHeight / 2)); /* bottom */
271
272 /* repeated Distance from default 140 x 200 Hoop */
273 /* TODO: Actually should be distance to custom hoop */
274 binaryWriteInt(file, (int) (630 - designWidth / 2)); /* left */
275 binaryWriteInt(file, (int) (550 - designHeight / 2)); /* top */
276 binaryWriteInt(file, (int) (630 - designWidth / 2)); /* right */
277 binaryWriteInt(file, (int) (550 - designHeight / 2)); /* bottom */
278
279 for (i = 0; i < pattern->thread_list->count; i++) {
280 int j = embThread_findNearestThread(pattern->thread_list->thread[i].color, (EmbThread *)jefThreads, 79);
281 binaryWriteInt(file, j);
282 }
283
284 for (i = 0; i < (minColors - colorlistSize); i++) {
285 int a = 0x0D;
286 binaryWriteInt(file, a);
287 }
288
289 pos.x = 0.0;
290 pos.y = 0.0;
291 for (i = 0; i < pattern->stitch_list->count; i++) {
292 unsigned char b[4];
293 EmbStitch st;
294 char dx, dy;
295 b[0] = 0;
296 b[1] = 0;
297 b[2] = 0;
298 b[3] = 0;
299 st = pattern->stitch_list->stitch[i];
300 dx = (char)emb_round(10.0*(st.x - pos.x));
301 dy = (char)emb_round(10.0*(st.y - pos.y));
302 pos.x += 0.1*dx;
303 pos.y += 0.1*dy;
304 jefEncode(b, dx, dy, st.flags);
305 if ((b[0] == 0x80) && ((b[1] == 1) || (b[1] == 2) || (b[1] == 4) || (b[1] == 0x10))) {
306 fwrite(b, 1, 4, file);
307 } else {
308 fwrite(b, 1, 2, file);
309 }
310 }
311 return 1;
312}
