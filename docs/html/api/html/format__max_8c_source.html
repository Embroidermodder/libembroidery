47char
48readMax(EmbPattern* pattern, FILE* file)
49{
50 unsigned char b[8];
51
52 fseek(file, 0xD5, SEEK_SET);
53 /* embInt_read(file, "stitchCount", &stitchCount, EMB_INT32_LITTLE); CHECK IF THIS IS PRESENT */
54 /* READ STITCH RECORDS */
55 while (fread(b, 1, 8, file) == 8) {
56 EmbReal dx, dy;
57 int flags;
58 flags = NORMAL;
59 dx = pfaffDecode(b[0], b[1], b[2]);
60 dy = pfaffDecode(b[4], b[5], b[6]);
61 embPattern_addStitchAbs(pattern, dx / 10.0, dy / 10.0, flags, 1);
62 }
63 embPattern_flipVertical(pattern);
64 return 1;
65}

67char
68writeMax(EmbPattern* pattern, FILE* file)
69{
70 int i;
71 EmbReal x, y;
72 EmbStitch st;
73
74 fwrite(max_header, 1, 0xD5, file);
75 for (i = 0; i < pattern->stitch_list->count; i++) {
76 st = pattern->stitch_list->stitch[i];
77 x = (int)emb_round(st.x * 10.0);
78 y = (int)emb_round(st.y * 10.0);
79 write_24bit(file, x);
80 write_24bit(file, y);
81 }
82 return 1;
