25{

28
29 embInt_read(file, "stitchCount", &stitchCount, EMB_INT16_LITTLE);
30 while (fread(data, 1, 3, file) == 3) {
31 int x = decodeNewStitch(data[0]);
32 int y = decodeNewStitch(data[1]);
33 int flag = NORMAL;
34 char val = data[2];
35 if (data[2] & 0x40) {
36 x = -x;
37 }
38 if (data[2] & 0x20) {
39 y = -y;
40 }
41 if (data[2] & 0x10) {
42 flag = TRIM;
43 }
44 if (data[2] & 0x01) {
45 flag = JUMP;
46 }
47 if ((val & 0x1E) == 0x02) {
48 flag = STOP;

50 /* Unknown values, possibly TRIM
51 155 = 1001 1011 = 0x9B
52 145 = 1001 0001 = 0x91
53 */

57 }*/
58 embPattern_addStitchRel(pattern, x / 10.0, y / 10.0, flag, 1);
59 }
60
61 return 1;
62}
