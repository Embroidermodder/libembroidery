24char
25readPcs(EmbPattern* pattern, const char* fileName, FILE* file)
26{
27 char allZeroColor = 1;
28 int i = 0;
29 unsigned char b[9];
30 EmbReal dx = 0, dy = 0;
31 int flags = 0, st = 0;
32 unsigned char version, hoopSize;
33 unsigned short colorCount;
34
35 version = (char)fgetc(file);
36 hoopSize = (char)fgetc(file); /* 0 for PCD, 1 for PCQ (MAXI), 2 for PCS with small hoop(80x80), */
37 /* and 3 for PCS with large hoop (115x120) */
38
39 switch(hoopSize) {
40 case 2:
41 pattern->hoop_width = 80.0;
42 pattern->hoop_height = 80.0;
43 break;
44 case 3:
45 pattern->hoop_width = 115.0;
46 pattern->hoop_height = 120.0;
47 break;
48 }
49
50 colorCount = fread_uint16(file);
51 if (emb_verbose>1) {
52 printf("version: %d\n", version);
53 printf("hoop size: %d\n", hoopSize);
54 printf("color count: %d\n", colorCount);
55 }
56
57 for (i = 0; i < colorCount; i++) {
58 EmbThread t;
59 embColor_read(file, &(t.color), 4);
60 strcpy(t.catalogNumber, "");
61 strcpy(t.description, "");
62 if (t.color.r || t.color.g || t.color.b) {
63 allZeroColor = 0;
64 }
65 embPattern_addThread(pattern, t);
66 }
67 if (allZeroColor) {
68 embPattern_loadExternalColorFile(pattern, fileName);
69 }
70 st = fread_uint16(file);
71 /* READ STITCH RECORDS */
72 for (i = 0; i < st; i++) {
73 flags = NORMAL;
74 if (fread(b, 1, 9, file) != 9)
75 break;
76
77 if (b[8] & 0x01) {
78 flags = STOP;
79 }
80 else if (b[8] & 0x04) {
81 flags = TRIM;
82 }
83 else if (b[8] != 0) {
84 /* TODO: ONLY INTERESTED IN THIS CASE TO LEARN MORE ABOUT THE FORMAT */
85 }
86 dx = pfaffDecode(b[1], b[2], b[3]);
87 dy = pfaffDecode(b[5], b[6], b[7]);
88 embPattern_addStitchAbs(pattern, dx / 10.0, dy / 10.0, flags, 1);
89 }
90
91 return 1;
92}

104 embColor_write(file, color, 4);

106
107 /* write remaining colors to reach 16 */

109

113 EmbStitch st = pattern->stitch_list->stitch[i];
114 pfaffEncode(file, (int)emb_round(st.x * 10.0), (int)emb_round(st.y * 10.0), st.flags);
115 }
116 return 1;
