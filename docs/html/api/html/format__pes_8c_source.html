47char
48readPes(EmbPattern* pattern, const char *fileName, FILE* file)
49{
50 int pecstart, numColors, x, version, i;
51 char signature[9];
52 fread(signature, 1, 8, file);
53 signature[8] = 0;
54 embInt_read(file, "pecstart", &pecstart, EMB_INT32_LITTLE);
55
56 version = 0;
57 for (i=0; i<N_PES_VERSIONS; i++) {
58 if (!strcmp(signature, pes_version_strings[i])) {
59 version = i;
60 break;
61 }
62 }
63
64 if (version >= PES0040) {
65 fseek(file, 4, SEEK_CUR);
66 readDescriptions(file, pattern);
67 }
68
69 switch (version) {
70 case PES0100:
71 readPESHeaderV10(file, pattern);
72 break;
73 case PES0090:
74 readPESHeaderV9(file, pattern);
75 break;
76 case PES0080:
77 readPESHeaderV8(file, pattern);
78 break;
79 case PES0070:
80 readPESHeaderV7(file, pattern);
81 break;
82 case PES0060:
83 readPESHeaderV6(file, pattern);
84 break;
85 case PES0056:
86 case PES0055:
87 case PES0050:
88 readPESHeaderV5(file, pattern);
89 break;
90 default:
91 break;
92 }
93
94 if (emb_verbose>1) {
95 printf("debug information for reading fileName: %s\n", fileName);
96 printf("pecstart = %d\n", pecstart);
97 }
98
99 /* fseek(file, pecstart + 48, SEEK_SET);
100 * This seems wrong based on the readPESHeader functions. */
101 fseek(file, pecstart, SEEK_SET);
102
103 numColors = fgetc(file) + 1;
104 for (x = 0; x < numColors; x++) {
105 int color_index = fgetc(file);
106 if (color_index >= pecThreadCount) {
107 color_index = 0;
108 }
109 embPattern_addThread(pattern, pecThreads[color_index]);
110 }
111
112 fseek(file, pecstart + 528, SEEK_SET);
113 readPecStitches(pattern, file);
114
115 embPattern_flipVertical(pattern);
116
117 return 1;
118}

337void
338pesWriteSewSegSection(EmbPattern* pattern, FILE* file) {
339 /* TODO: pointer safety */
340 short* colorInfo = 0;
341 int flag = 0;
342 int count = 0;
343 int colorCode = -1;
344 int stitchType = 0;
345 int blockCount = 0;
346 int colorCount = 0;
347 int newColorCode = 0;
348 int colorInfoIndex = 0;
349 int i, j;
350 EmbRect bounds = embPattern_calcBoundingBox(pattern);
351
352 for (i = 0; i < pattern->stitch_list->count; i++) {
353 EmbColor color;
354 EmbStitch st = pattern->stitch_list->stitch[i];
355 flag = st.flags;
356 if (st.color < pattern->thread_list->count) {
357 color = pattern->thread_list->thread[st.color].color;
358 }
359 else {
360 color = pecThreads[0].color;
361 }
362 newColorCode = embThread_findNearestThread(color, (EmbThread*)pecThreads, pecThreadCount);
363 if (newColorCode != colorCode) {
364 colorCount++;
365 colorCode = newColorCode;
366 }
367 while (i < pattern->stitch_list->count && (flag == st.flags)) {
368 st = pattern->stitch_list->stitch[i];
369 count++;
370 i++;
371 }
372 blockCount++;
373 }
374
375 binaryWriteShort(file, (short)blockCount); /* block count */
376 binaryWriteUShort(file, 0xFFFF);
377 binaryWriteShort(file, 0x00);
378
379 binaryWriteShort(file, 0x07); /* string length */
380 fwrite("CSewSeg", 1, 7, file);
381
382 if (colorCount > 1000) {
383 puts("Color count exceeds 1000 this is likely an error. Truncating to 1000.");
384 colorCount = 1000;
385 }
386
387 colorInfo = (short *) calloc(colorCount * 2, sizeof(short));
388 colorCode = -1;
389 blockCount = 0;
390 for (i = 0; i < pattern->stitch_list->count; i++) {
391 EmbColor color;
392 EmbStitch st;
393 st = pattern->stitch_list->stitch[i];
394 j = i;
395 flag = st.flags;
396 color = pattern->thread_list->thread[st.color].color;
397 newColorCode = embThread_findNearestThread(color, (EmbThread*)pecThreads, pecThreadCount);
398 if (newColorCode != colorCode) {
399 if (colorInfoIndex+2 > colorCount) {
400 puts("Ran out of memory for color info.");
401 break;
402 }
403 colorInfo[colorInfoIndex++] = (short)blockCount;
404 colorInfo[colorInfoIndex++] = (short)newColorCode;
405 colorCode = newColorCode;
406 }
407 count = 0;
408 while (j < pattern->stitch_list->count && (flag == st.flags)) {
409 st = pattern->stitch_list->stitch[j];
410 count++;
411 j++;
412 }
413 if (flag & JUMP) {
414 stitchType = 1;
415 }
416 else {
417 stitchType = 0;
418 }
419
420 binaryWriteShort(file, (short)stitchType); /* 1 for jump, 0 for normal */
421 binaryWriteShort(file, (short)colorCode); /* color code */
422 binaryWriteShort(file, (short)count); /* stitches in block */
423 j = i;
424 while (j < pattern->stitch_list->count && (flag == st.flags)) {
425 st = pattern->stitch_list->stitch[j];
426 binaryWriteShort(file, (short)(st.x - bounds.left));
427 binaryWriteShort(file, (short)(st.y + bounds.top));
428 }
429 if (j < pattern->stitch_list->count ) {
430 binaryWriteUShort(file, 0x8003);
431 }
432 blockCount++;
433 i = j;
434 }
435 binaryWriteShort(file, (short)colorCount);
436 for (i = 0; i < colorCount; i++) {
437 binaryWriteShort(file, colorInfo[i * 2]);
438 binaryWriteShort(file, colorInfo[i * 2 + 1]);
439 }
440 binaryWriteInt(file, 0);
441 safe_free(colorInfo);
442}

444void
445pesWriteEmbOneSection(EmbPattern* pattern, FILE* file) {
446 /* TODO: pointer safety */
447 float x, width, height;
448 int hoopHeight = 1800, hoopWidth = 1300;
449 EmbRect bounds;
450 binaryWriteShort(file, 0x07); /* string length */
451 fwrite("CEmbOne", 1, 7, file);
452 bounds = embPattern_calcBoundingBox(pattern);
453
454 fpad(file, 0, 16);
455
456 /* AffineTransform */
457 x = 1.0;
458 embInt_write(file, ".", &x, EMB_INT32_LITTLE);
459 x = 0.0;
460 embInt_write(file, ".", &x, EMB_INT32_LITTLE);
461 x = 0.0;
462 embInt_write(file, ".", &x, EMB_INT32_LITTLE);
463 x = 1.0;
464 embInt_write(file, ".", &x, EMB_INT32_LITTLE);
465 width = bounds.right - bounds.left;
466 height = bounds.bottom - bounds.top;
467 x = (float)((width - hoopWidth) / 2);
468 embInt_write(file, ".", &x, EMB_INT32_LITTLE);
469 x = (float)((height + hoopHeight) / 2);
470 embInt_write(file, ".", &x, EMB_INT32_LITTLE);
471
472 binaryWriteShort(file, 1);
473 binaryWriteShort(file, 0); /* Translate X */
474 binaryWriteShort(file, 0); /* Translate Y */
475 binaryWriteShort(file, (short)width);
476 binaryWriteShort(file, (short)height);
477
478 fpad(file, 0, 8);
479 /*WriteSubObjects(br, pes, SubBlocks); */
480}

482char
483writePes(EmbPattern* pattern, const char *fileName, FILE* file)
484{
485 int pecLocation;
486 embPattern_flipVertical(pattern);
487 embPattern_scale(pattern, 10.0);
488 fwrite("#PES0001", 1, 8, file);
489 /* WRITE PECPointer 32 bit int */
490 binaryWriteInt(file, 0x00);
491
492 binaryWriteShort(file, 0x01);
493 binaryWriteShort(file, 0x01);
494
495 /* Write object count */
496 binaryWriteShort(file, 0x01);
497 binaryWriteUShort(file, 0xFFFF); /* command */
498 binaryWriteShort(file, 0x00); /* unknown */
499
500 pesWriteEmbOneSection(pattern, file);
501 pesWriteSewSegSection(pattern, file);
502
503 pecLocation = ftell(file);
504 fseek(file, 0x08, SEEK_SET);
505 fputc((unsigned char)(pecLocation & 0xFF), file);
506 fputc((unsigned char)(pecLocation >> 8) & 0xFF, file);
507 fputc((unsigned char)(pecLocation >> 16) & 0xFF, file);
508 fseek(file, 0x00, SEEK_END);
509 writePecStitches(pattern, file, fileName);
510 return 1;
511}
