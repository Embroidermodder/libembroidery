33char
34readSew(EmbPattern* pattern, FILE* file)
35{
36 int i, flags, numberOfColors, fileLength;
37 char dx, dy, thisStitchIsJump = 0;
38
39 fseek(file, 0x00, SEEK_END);
40 fileLength = ftell(file);
41 fseek(file, 0x00, SEEK_SET);
42 numberOfColors = fgetc(file);
43 numberOfColors += (fgetc(file) << 8);
44
45
46 for (i = 0; i < numberOfColors; i++) {
47 int color = fread_int16(file);
48 embPattern_addThread(pattern, jefThreads[color%78]);
49 }
50 fseek(file, 0x1D78, SEEK_SET);
51
52 for (i = 0; ftell(file) < fileLength; i++) {
53 unsigned char b[2];
54 fread(b, 1, 2, file);
55
56 flags = NORMAL;
57 if (thisStitchIsJump) {
58 flags = TRIM;
59 thisStitchIsJump = 0;
60 }
61 if (b[0] == 0x80) {
62 if (b[1] == 1) {
63 fread(b, 1, 2, file);
64 flags = STOP;
65 } else if ((b[1] == 0x02) || (b[1] == 0x04)) {
66 thisStitchIsJump = 1;
67 fread(b, 1, 2, file);
68 flags = TRIM;
69 } else if (b[1] == 0x10) {
70 break;
71 }
72 }
73 dx = sewDecode(b[0]);
74 dy = sewDecode(b[1]);
75 if (abs(dx) == 127 || abs(dy) == 127) {
76 thisStitchIsJump = 1;
77 flags = TRIM;
78 }
79 embPattern_addStitchRel(pattern, dx / 10.0, dy / 10.0, flags, 1);
80 }
81 printf("current position: %ld\n", ftell(file));
82 return 1;
83}

85char
86writeSew(EmbPattern* pattern, FILE* file)
87{
88 int i;
89 EmbReal xx = 0.0, yy = 0.0;
90 binaryWriteShort(file, pattern->thread_list->count);
91
92 if (emb_verbose>1) {
93 printf("Debugging Information\n");
94 printf("number of colors = %d\n", pattern->thread_list->count);
95 printf("number of stitches = %d\n", pattern->stitch_list->count);
96 }
97
98 for (i = 0; i < pattern->thread_list->count; i++) {
99 short thr;
100 EmbColor col;
101 col = pattern->thread_list->thread[i].color;
102 thr = embThread_findNearestThread(col, (EmbThread *)jefThreads, 79);
103 binaryWriteShort(file, thr);
104 }
105 fpad(file, 0, 0x1D78 - 2 - pattern->thread_list->count * 2);
106
107 for (i = 0; i < pattern->stitch_list->count; i++) {
108 EmbStitch st;
109 unsigned char b[4];
110 char dx, dy;
111 st = pattern->stitch_list->stitch[i];
112 dx = (char)emb_round(10.0*(st.x - xx));
113 dy = (char)emb_round(10.0*(st.y - yy));
114 xx += 0.1*dx;
115 yy += 0.1*dy;
116 if (st.flags & STOP) {
117 b[0] = 0x80;
118 b[1] = 0x01;
119 b[2] = dx;
120 b[3] = dy;
121 fwrite(b, 1, 4, file);
122 } else if (st.flags & END) {
123 b[0] = 0x80;
124 b[1] = 0x10;
125 b[2] = 0;
126 b[3] = 0;
127 fwrite(b, 1, 4, file);
128 } else if ((st.flags & TRIM) || (st.flags & JUMP)) {
129 b[0] = 0x80;
130 b[1] = 2;
131 b[2] = dx;
132 b[3] = dy;
133 fwrite(b, 1, 4, file);
134 } else {
135 b[0] = dx;
136 b[1] = dy;
137 fwrite(b, 1, 2, file);
138 }
139 }
140 return 1;
141}
