23char
24readSst(EmbPattern* pattern, FILE* file)
25{
26 int fileLength;
27
28 fseek(file, 0, SEEK_END);
29 fileLength = ftell(file);
30 fseek(file, 0xA0, SEEK_SET); /* skip the all zero header */
31 while (ftell(file) < fileLength) {
32 int stitchType = NORMAL;
33
34 int b1 = fgetc(file);
35 int b2 = fgetc(file);
36 unsigned char commandByte = (unsigned char)fgetc(file);
37
38 if (commandByte == 0x04) {
39 embPattern_addStitchRel(pattern, 0, 0, END, 1);
40 break;
41 }
42
43 if ((commandByte & 0x01) == 0x01)
44 stitchType = STOP;
45 if ((commandByte & 0x02) == 0x02)
46 stitchType = JUMP;
47 if ((commandByte & 0x10) != 0x10)
48 b2 = -b2;
49 if ((commandByte & 0x40) == 0x40)
50 b1 = -b1;
51 embPattern_addStitchRel(pattern, b1 / 10.0, b2 / 10.0, stitchType, 1);
52 }
53
54 return 1; /*TODO: finish readSst */
55}
