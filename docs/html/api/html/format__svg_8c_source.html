38EmbColor svgColorToEmbColor(char* colorString)
39{
40 EmbColor c;
41 char* pEnd = 0;
42 char* colorStr = copy_trim(colorString); /* Trim out any junk spaces */
43 int length = strlen(colorStr);
44 int tableColor;
45
46 /* SVGTiny1.2 Spec Section 11.13.1 syntax for color values */
47 if (length == 7 && colorStr[0] == '#') /* Six digit hex — #rrggbb */ {
48 while (*colorStr=='#' && *colorStr!=0) {
49 colorStr++;
50 }
51 c = embColor_fromHexStr(colorStr);
52 }
53 else if (length == 4 && colorStr[0] == '#') /* Three digit hex — #rgb */ {
54 /* Convert the 3 digit hex to a six digit hex */
55 char hex[7];
56 sprintf(hex, "%c%c%c%c%c%c", colorStr[1], colorStr[1], colorStr[2],
57 colorStr[2], colorStr[3], colorStr[3]);
58
59 c = embColor_fromHexStr(hex);
60 }
61 else if (strstr(colorStr, "%")) /* Float functional — rgb(R%, G%, B%) */ {
62 char *s = colorStr;
63 /* replace characters we aren't using with spaces */
64 for (; *s; s++) {
65 if (*s == 'r' || *s == 'g' || *s == 'b' || *s == ','
66 || *s == '(' || *s == ')' || *s == '%') {
67 *s = ' ';
68 }
69 }
70 c.r = (unsigned char)emb_round(255.0/100.0 * strtod(colorStr, &pEnd));
71 c.g = (unsigned char)emb_round(255.0/100.0 * strtod(pEnd, &pEnd));
72 c.b = (unsigned char)emb_round(255.0/100.0 * strtod(pEnd, &pEnd));
73 }
74 else if (length > 3 && colorStr[0] == 'r' && colorStr[1] == 'g' && colorStr[2] == 'b') {
75 /* Integer functional — rgb(rrr, ggg, bbb) */
76 char *s = colorStr;
77 /* replace characters we aren't using with spaces */
78 for (; *s; s++) {
79 if (*s == 'r' || *s == 'g' || *s == 'b' || *s == ','
80 || *s == '(' || *s == ')') {
81 *s = ' ';
82 }
83 }
84 c.r = (unsigned char)strtol(colorStr, &pEnd, 10);
85 c.g = (unsigned char)strtol(pEnd, &pEnd, 10);
86 c.b = (unsigned char)strtol(pEnd, &pEnd, 10);
87 }
88 else /* Color keyword */ {
89 tableColor = threadColor(colorStr, SVG_Colors);
90 if (tableColor < 0) {
91 printf("SVG color string not found: %s.\n", colorStr);
92 c.r = 0;
93 c.g = 0;
94 c.b = 0;
95 }
96 else {
97 c.r = (tableColor >> 16) & 0xFF;
98 c.g = (tableColor >> 8) & 0xFF;
99 c.b = tableColor & 0xFF;
100 }
101 }
102
103 safe_free(colorStr);
104 /* Returns black if all else fails */
105 return c;
106}

186 }
187 else {
188 EmbLine line;
189 line.start.x = atof(x1);
190 line.start.y = atof(y1);
191 line.end.x = atof(x2);
192 line.end.y = atof(y2);
193 embPattern_addLineAbs(p, line);
194 }

656 if (stringInArray(buff, svg_element_tokens)) {
657 return SVG_ELEMENT;
658 }
659 /* Attempt to identify the program that created the SVG file.
660 * This should be in a comment at that occurs before the svg element. */
661 else if (!strcmp(buff, "Embroidermodder")) {
662 svgCreator = SVG_CREATOR_EMBROIDERMODDER;
663 }
664 else if (!strcmp(buff, "Illustrator")) {
665 svgCreator = SVG_CREATOR_ILLUSTRATOR;
666 }
667 else if (!strcmp(buff, "Inkscape")) {
668 svgCreator = SVG_CREATOR_INKSCAPE;
669 }
670
671 return SVG_NULL;
672}

692void
693svgProcess(int c, const char* buff)
694{
695 if (svgExpect == SVG_EXPECT_ELEMENT) {
696 char advance = 0;
697 if (buff[0] == '/') {
698 return;
699 }
700 advance = (char)svgIsElement(buff);
701 if (advance) {
702 printf("ELEMENT:\n");
703 svgExpect = SVG_EXPECT_ATTRIBUTE;
704 current_element_id = svg_identify_element((char*)buff);
705 } else {
706 return;
707 }
708 } else if (svgExpect == SVG_EXPECT_ATTRIBUTE) {
709 char advance = 0;
710 switch (current_element_id) {
711 case ELEMENT_A:
712 case ELEMENT_CIRCLE:
713 case ELEMENT_DEFS:
714 case ELEMENT_ELLIPSE:
715 case ELEMENT_FOREIGN_OBJECT:
716 case ELEMENT_G:
717 case ELEMENT_LINE:
718 case ELEMENT_LINEAR_GRADIENT:
719 case ELEMENT_PATH:
720 case ELEMENT_POLYGON:
721 case ELEMENT_POLYLINE:
722 case ELEMENT_RADIAL_GRADIENT:
723 case ELEMENT_RECT:
724 case ELEMENT_SOLID_COLOR:
725 case ELEMENT_STOP:
726 case ELEMENT_SVG:
727 case ELEMENT_SWITCH:
728 case ELEMENT_TEXT:
729 case ELEMENT_TEXT_AREA:
730 case ELEMENT_TSPAN:
731 case ELEMENT_USE:
732 if (stringInArray(buff, svg_property_tokens)) {
733 advance = SVG_PROPERTY;
734 }
735 break;
736 case ELEMENT_ANIMATION:
737 case ELEMENT_AUDIO:
738 case ELEMENT_DESC:
739 case ELEMENT_IMAGE:
740 case ELEMENT_METADATA:
741 case ELEMENT_TITLE:
742 case ELEMENT_VIDEO:
743 if (stringInArray(buff, svg_media_property_tokens)) {
744 advance = SVG_MEDIA_PROPERTY;
745 }
746 break;
747 default: break;
748 }
749 if (!advance) {
750 if (stringInArray(buff, (const char **)svg_attribute_table[current_element_id])) {
751 advance = SVG_ATTRIBUTE;
752 }
753 printf("ERROR %s not found in svg_attribute_table[%d].\n",
754 buff, current_element_id);
755 }
756 if (advance) {
757 printf("ATTRIBUTE:\n");
758 svgExpect = SVG_EXPECT_VALUE;
759 strcpy(currentAttribute, buff);
760 }
761 } else if (svgExpect == SVG_EXPECT_VALUE) {
762 int last = strlen(buff) - 1;
763 printf("VALUE:\n");
764
765 /* single-value */
766 if ((buff[0] == '"' || buff[0] == '\'') && (buff[last] == '/' || buff[last] == '"' || buff[last] == '\'') && !svgMultiValue) {
767 svgExpect = SVG_EXPECT_ATTRIBUTE;
768 strcpy(attributeList[n_attributes].name, currentAttribute);
769 strcpy(attributeList[n_attributes].value, buff);
770 n_attributes++;
