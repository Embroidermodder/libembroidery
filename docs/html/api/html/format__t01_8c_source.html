38char
39writeT01(EmbPattern* pattern, FILE* file)
40{
41 EmbRect boundingRect;
42 int i;
43 EmbVector pos;
44
45 embPattern_correctForMaxStitchLength(pattern, 12.1, 12.1);
46
47 boundingRect = embPattern_calcBoundingBox(pattern);
48 if (emb_verbose>1) {
49 printf("bounding rectangle with top %f not used ", boundingRect.top);
50 printf("in the function writeT01\n");
51 }
52 pos.x = 0.0;
53 pos.y = 0.0;
54 for (i = 0; i < pattern->stitch_list->count; i++) {
55 unsigned char b[3];
56 int dx, dy;
57 EmbStitch st = pattern->stitch_list->stitch[i];
58 /* convert from mm to 0.1mm for file format */
59 dx = (int)emb_round(10.0*(st.x - pos.x));
60 dy = (int)emb_round(10.0*(st.y - pos.y));
61 pos.x += 0.1*dx;
62 pos.y += 0.1*dy;
63 encode_t01_record(b, dx, dy, st.flags);
64 fwrite(b, 1, 3, file);
65 }
66 return 1;
67}
