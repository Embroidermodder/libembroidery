44char
45readThr(EmbPattern* pattern, FILE* file)
46{
47 ThredHeader header;
48 EmbColor background;
49 int currentColor;
50 int i;
51
52 embInt_read(file, "sig version", &(header.sigVersion), EMB_INT32_LITTLE);
53 embInt_read(file, "length", &(header.length), EMB_INT32_LITTLE);
54 embInt_read(file, "number of stitches", &(header.numStiches), EMB_INT16_LITTLE);
55 header.hoopSize = fread_uint16(file);
56 for (i=0; i<7; i++) {
57 header.reserved[i] = fread_uint16(file);
58 }
59
60 if ((header.sigVersion & 0xffffff) == 0x746872) {
61 unsigned int verVar = (header.sigVersion & 0xff000000) >> 24;
62 switch (verVar) {
63 case 0:
64 break;
65 case 1:
66 case 2:
67 /* skip the file header extension */
68 fseek(file, 144, SEEK_CUR);
69 break;
70 default:
71 return 0; /* unsupported version */
72 }
73 }
74 currentColor = -1;
75 for (i = 0; i < header.numStiches; i++) {
76 int type = NORMAL;
77 float x, y;
78 unsigned int color;
79 embInt_read(file, "x", &x, EMB_INT32_LITTLE);
80 embInt_read(file, "y", &y, EMB_INT32_LITTLE);
81 embInt_read(file, "color", &color, EMB_INT32_LITTLE);
82 x /= 10.0;
83 y /= 10.0;
84
85 if ((int)(color & 0xF) != currentColor) {
86 currentColor = (int)color & 0xF;
87 embPattern_changeColor(pattern, currentColor);
88 type = STOP | TRIM;
89 }
90 embPattern_addStitchAbs(pattern, x, y, type, 0);
91 }
92 fseek(file, 16, SEEK_CUR); /* skip bitmap name (16 chars) */
93
94 embColor_read(file, &background, 4);
95 if (emb_verbose>1) {
96 printf("background: %c %c %c\n", background.r, background.g, background.b);
97 }
98 for (i = 0; i < 16; i++) {
99 EmbThread thread;
100 strcpy(thread.description, "NULL");
101 strcpy(thread.catalogNumber, "NULL");
102 embColor_read(file, &(thread.color), 4);
103 embPattern_addThread(pattern, thread);
104 }
105 /* 64 bytes of rgbx(4 bytes) colors (16 custom colors) */
106 /* 16 bytes of thread size (ascii representation ie. '4') */
107 return 1;
108}

110char
111writeThr(EmbPattern* pattern, FILE* file)
112{
113 int i, stitchCount;
114 unsigned char version = 0;
115 ThredHeader header;
116 ThredExtension extension;
117 char bitmapName[16];
118
119 stitchCount = pattern->stitch_list->count;
120
121 memset(&header, 0, sizeof(ThredHeader));
122 header.sigVersion = 0x746872 | (version << 24);
123 header.length = stitchCount * 12 + 16;
124 if (version == 1 || version == 2) {
125 header.length = header.length + sizeof(ThredHeader);
126 }
127 header.numStiches = (unsigned short)stitchCount; /* number of stitches in design */
128 header.hoopSize = 5;
129
130 binaryWriteUInt(file, header.sigVersion);
131 binaryWriteUInt(file, header.length);
132 binaryWriteUShort(file, header.numStiches);
133 binaryWriteUShort(file, header.hoopSize);
134 for (i=0; i<7; i++) {
135 binaryWriteUShort(file, header.reserved[i]);
136 }
137
138 if (version == 1 || version == 2) {
139 memset(&extension, 0, sizeof(ThredExtension));
140 extension.auxFormat = 1;
141 extension.hoopX = 640;
142 extension.hoopY = 640;
143
144 embInt_write(file, ".", &(extension.hoopX), EMB_INT32_LITTLE);
145 embInt_write(file, ".", &(extension.hoopY), EMB_INT32_LITTLE);
146 embInt_write(file, ".", &(extension.stitchGranularity), EMB_INT32_LITTLE);
147 fwrite(extension.creatorName, 1, 50, file);
148 fwrite(extension.modifierName, 1, 50, file);
149 fputc(extension.auxFormat, file);
150 fwrite(extension.reserved, 1, 31, file);
151 }
152
153 /* write stitches */
154 for (i = 0; i < pattern->stitch_list->count; i++) {
155 unsigned int NOTFRM = 0x00080000;
156 EmbStitch st = pattern->stitch_list->stitch[i];
157 float x, y;
158 x = (float)(st.x * 10.0);
159 y = (float)(st.y * 10.0);
160 embInt_write(file, ".", &x, EMB_INT32_LITTLE);
161 embInt_write(file, ".", &y, EMB_INT32_LITTLE);
162 binaryWriteUInt(file, NOTFRM | (st.color & 0x0F));
163 }
164 fwrite(bitmapName, 1, 16, file);
165 /* background color */
166 fwrite("\xFF\xFF\xFF\x00", 1, 4, file);
167
168 for (i = 0; i < pattern->thread_list->count; i++) {
169 EmbColor c = pattern->thread_list->thread[i].color;
170 embColor_write(file, c, 4);
171 if (i >= 16) break;
172 }
173
174 /* write remaining colors if not yet 16 */
175 fpad(file, 0, 4*(16-i));
176
177 fpad(file, '4', 16);
178 return 1;
179}
