23char
24readTxt(EmbPattern* pattern, FILE* file)
25{
26 char line[100];
27 int stated_count, i;
28 emb_readline(file, line, 99);
29 stated_count = atoi(line);
30 for (i=0; i<stated_count; i++) {
31 EmbStitch st;
32 char *p, *y, *color, *flags;
33 int state = 0;
34 emb_readline(file, line, 99);
35 p = line;
36 y = line;
37 color = line;
38 flags = line;
39 for (p=line; *p; p++) {
40 if (*p == ',' || *p == ' ') {
41 *p = 0;
42 switch (state) {
43 case 0:
44 y = p+1;
45 break;
46 case 1:
47 color = p+7;
48 break;
49 case 2:
50 flags = p+7;
51 break;
52 default:
53 break;
54 }
55 state++;
56 }
57 }
58 st.x = atof(line);
59 st.y = atof(y);
60 st.color = atoi(color);
61 st.flags = atoi(flags);
62 embPattern_addStitchAbs(pattern, st.x, st.y, st.flags, st.color);
63 }
64 puts("Overridden, defaulting to dst.");
65 readDst(pattern, file);
66 return 0; /*TODO: finish readTxt */
67}
