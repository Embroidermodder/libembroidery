23char
24readU00(EmbPattern* pattern, FILE* file)
25{
26 int i;
27 char dx = 0, dy = 0;
28 int flags = NORMAL;
29 unsigned char b[3];
30
31 if (!check_header_present(file, 0x100)) {
32 return 0;
33 }
34
35 /* 16 3byte RGB's start @ 0x08 followed by 14 bytes between
36 0 and 15 with index of color for each color change */
37 fseek(file, 0x08, SEEK_SET);
38 for (i = 0; i < 16; i++) {
39 EmbThread t;
40 embColor_read(file, &(t.color), 3);
41 strcpy(t.catalogNumber, "");
42 strcpy(t.description, "");
43 embPattern_addThread(pattern, t);
44 }
45
46 fseek(file, 0x100, SEEK_SET);
47 while (fread(b, 1, 3, file) == 3) {
48 char negativeX , negativeY;
49
50 if (b[0] == 0xF8 || b[0] == 0x87 || b[0] == 0x91) {
51 break;
52 }
53 if ((b[0] & 0x0F) == 0) {
54 flags = NORMAL;
55 } else if ((b[0] & 0x1f) == 1) {
56 flags = JUMP;
57 } else if ((b[0] & 0x0F) > 0) {
58 flags = STOP;
59 }
60 negativeX = ((b[0] & 0x20) > 0);
61 negativeY = ((b[0] & 0x40) > 0);
62
63 dx = (char) b[2];
64 dy = (char) b[1];
65 if (negativeX) {
66 dx = (char) -dx;
67 }
68 if (negativeY) {
69 dy = (char) -dy;
70 }
71 embPattern_addStitchRel(pattern, dx / 10.0, dy / 10.0, flags, 1);
72 }
73 return 1;
74}
