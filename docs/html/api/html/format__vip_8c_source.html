88char
89readVip(EmbPattern* pattern, FILE* file)
90{
91 int fileLength;
92 int i;
93 unsigned char prevByte = 0;
94 unsigned char *attributeData = 0, *decodedColors = 0;
95 unsigned char *attributeDataDecompressed = 0;
96 unsigned char *xData = 0, *xDecompressed = 0, *yData = 0, *yDecompressed = 0;
97 VipHeader header;
98
99 fileLength = check_header_present(file, 32);
100 if (!fileLength) {
101 printf("ERROR: file shorter than header.");
102 return 0;
103 }
104
105 embInt_read(file, "magicCode", &(header.magicCode), EMB_INT32_LITTLE);
106 embInt_read(file, "numberOfStitches", &(header.numberOfStitches), EMB_INT32_LITTLE);
107 embInt_read(file, ".", &(header.numberOfColors), EMB_INT32_LITTLE);
108
109 embInt_read(file, ".", &(header.postitiveXHoopSize), EMB_INT16_LITTLE);
110 embInt_read(file, ".", &(header.postitiveYHoopSize), EMB_INT16_LITTLE);
111 embInt_read(file, ".", &(header.negativeXHoopSize), EMB_INT16_LITTLE);
112 embInt_read(file, ".", &(header.negativeYHoopSize), EMB_INT16_LITTLE);
113 embInt_read(file, ".", &(header.attributeOffset), EMB_INT32_LITTLE);
114 embInt_read(file, ".", &(header.xOffset), EMB_INT32_LITTLE);
115 embInt_read(file, ".", &(header.yOffset), EMB_INT32_LITTLE);
116
117 fread(header.stringVal, 1, 8, file); /* TODO: check return value */
118
119 embInt_read(file, ".", &(header.unknown), EMB_INT16_LITTLE);
120
121 embInt_read(file, ".", &(header.colorLength), EMB_INT32_LITTLE);
122 decodedColors = (unsigned char*)malloc(header.numberOfColors*4);
123 if (!decodedColors) {
124 printf("ERROR: format-vip.c readVip(), ");
125 printf("cannot allocate memory for decodedColors\n");
126 return 0;
127 }
128 for (i = 0; i < header.numberOfColors*4; ++i) {
129 unsigned char inputByte = (char)fgetc(file);
130 unsigned char tmpByte = (unsigned char) (inputByte ^ vipDecodingTable[i]);
131 decodedColors[i] = (unsigned char) (tmpByte ^ prevByte);
132 prevByte = inputByte;
133 }
134 for (i = 0; i < header.numberOfColors; i++) {
135 EmbThread thread;
136 int startIndex = i << 2;
137 thread.color.r = decodedColors[startIndex];
138 thread.color.g = decodedColors[startIndex + 1];
139 thread.color.b = decodedColors[startIndex + 2];
140 /* printf("%d\n", decodedColors[startIndex + 3]); */
141 embPattern_addThread(pattern, thread);
142 }
143 fseek(file, header.attributeOffset, SEEK_SET);
144 attributeData = (unsigned char*)malloc(header.xOffset - header.attributeOffset);
145 if (!attributeData) {
146 printf("ERROR: format-vip.c readVip(), cannot allocate memory for attributeData\n");
147 return 0;
148 }
149 fread(attributeData, 1, header.xOffset - header.attributeOffset, file); /* TODO: check return value */
150 attributeDataDecompressed = vipDecompressData(attributeData, header.xOffset - header.attributeOffset, header.numberOfStitches);
151
152 fseek(file, header.xOffset, SEEK_SET);
153 xData = (unsigned char*)malloc(header.yOffset - header.xOffset);
154 if (!xData) {
155 printf("ERROR: format-vip.c readVip(), cannot allocate memory for xData\n");
156 return 0;
157 }
158 fread(xData, 1, header.yOffset - header.xOffset, file); /* TODO: check return value */
159 xDecompressed = vipDecompressData(xData, header.yOffset - header.xOffset, header.numberOfStitches);
160
161 fseek(file, header.yOffset, SEEK_SET);
162 yData = (unsigned char*)malloc(fileLength - header.yOffset);
163 if (!yData) { printf("ERROR: format-vip.c readVip(), cannot allocate memory for yData\n"); return 0; }
164 fread(yData, 1, fileLength - header.yOffset, file); /* TODO: check return value */
165 yDecompressed = vipDecompressData(yData, fileLength - header.yOffset, header.numberOfStitches);
166
167 for (i = 0; i < header.numberOfStitches; i++) {
168 embPattern_addStitchRel(pattern,
169 vipDecodeByte(xDecompressed[i]) / 10.0,
170 vipDecodeByte(yDecompressed[i]) / 10.0,
171 vipDecodeStitchType(attributeDataDecompressed[i]), 1);
172 }
173 embPattern_addStitchRel(pattern, 0, 0, END, 1);
174
175 safe_free(attributeData);
176 safe_free(xData);
177 safe_free(yData);
178 safe_free(attributeDataDecompressed);
179 safe_free(xDecompressed);
180 safe_free(yDecompressed);
181 return 1;
182}

220char
221writeVip(EmbPattern* pattern, FILE* file)
222{
223 EmbRect boundingRect;
224 int stitchCount, minColors, patternColor;
225 int attributeSize = 0;
226 int xCompressedSize = 0;
227 int yCompressedSize = 0;
228 EmbReal previousX = 0;
229 EmbReal previousY = 0;
230 unsigned char* xValues = 0, *yValues = 0, *attributeValues = 0;
231 EmbReal xx = 0.0;
232 EmbReal yy = 0.0;
233 int flags = 0;
234 int i = 0;
235 unsigned char* attributeCompressed = 0, *xCompressed = 0;
236 unsigned char* yCompressed = 0, *decodedColors = 0, *encodedColors = 0;
237 unsigned char prevByte = 0;
238
239 puts("VIP not yet implemented.");
240 return 0;
241
242 stitchCount = pattern->stitch_list->count;
243 minColors = pattern->thread_list->count;
244 decodedColors = (unsigned char*)malloc(minColors << 2);
245 if (!decodedColors) {
246 return 0;
247 }
248 encodedColors = (unsigned char*)malloc(minColors << 2);
249 if (!encodedColors) {
250 safe_free(decodedColors);
251 return 0;
252 }
253 /* embPattern_correctForMaxStitchLength(pattern, 0x7F, 0x7F); */
254
255 patternColor = minColors;
256 if (emb_verbose>1) {
257 printf("patternColor: %d\n", patternColor);
258 }
259 if (minColors > 24) {
260 minColors = 24;
261 }
262 binaryWriteUInt(file, 0x0190FC5D);
263 binaryWriteUInt(file, stitchCount);
264 binaryWriteUInt(file, minColors);
265
266 boundingRect = embPattern_calcBoundingBox(pattern);
267 binaryWriteShort(file, (short) emb_round(boundingRect.right * 10.0));
268 binaryWriteShort(file, (short) -emb_round(boundingRect.top * 10.0 - 1.0));
269 binaryWriteShort(file, (short) emb_round(boundingRect.left * 10.0));
270 binaryWriteShort(file, (short) -emb_round(boundingRect.bottom * 10.0 - 1.0));
271
272 binaryWriteUInt(file, 0x38 + (minColors << 3));
273
274 xValues = (unsigned char*)malloc(sizeof(unsigned char)*(stitchCount));
275 yValues = (unsigned char*)malloc(sizeof(unsigned char)*(stitchCount));
276 attributeValues = (unsigned char*)malloc(sizeof(unsigned char)*(stitchCount));
277 if (xValues && yValues && attributeValues) {
278 for (i = 0; i < pattern->stitch_list->count; i++) {
279 EmbStitch st;
280 st = pattern->stitch_list->stitch[i];
281 xx = st.x;
282 yy = st.y;
283 flags = st.flags;
284 xValues[i] = vipEncodeByte((xx - previousX) * 10.0);
285 previousX = xx;
286 yValues[i] = vipEncodeByte((yy - previousY) * 10.0);
287 previousY = yy;
288 attributeValues[i] = vipEncodeStitchType(flags);
289 }
290 attributeCompressed = vipCompressData(attributeValues, stitchCount, &attributeSize);
291 xCompressed = vipCompressData(xValues, stitchCount, &xCompressedSize);
292 yCompressed = vipCompressData(yValues, stitchCount, &yCompressedSize);
293
294 binaryWriteUInt(file, (unsigned int) (0x38 + (minColors << 3) + attributeSize));
295 binaryWriteUInt(file, (unsigned int) (0x38 + (minColors << 3) + attributeSize + xCompressedSize));
296 binaryWriteUInt(file, 0x00000000);
297 binaryWriteUInt(file, 0x00000000);
298 binaryWriteUShort(file, 0x0000);
299
300 binaryWriteInt(file, minColors << 2);
301
302 for (i = 0; i < minColors; i++) {
303 int byteChunk = i << 2;
304 EmbColor currentColor = pattern->thread_list->thread[i].color;
305 decodedColors[byteChunk] = currentColor.r;
306 decodedColors[byteChunk + 1] = currentColor.g;
307 decodedColors[byteChunk + 2] = currentColor.b;
308 decodedColors[byteChunk + 3] = 0x01;
309 }
310
311 for (i = 0; i < minColors << 2; ++i) {
312 unsigned char tmpByte = (unsigned char) (decodedColors[i] ^ vipDecodingTable[i]);
313 prevByte = (unsigned char) (tmpByte ^ prevByte);
314 fputc(prevByte, file);
315 }
316 for (i = 0; i <= minColors; i++) {
317 binaryWriteInt(file, 1);
318 }
319 binaryWriteUInt(file, 0); /* string length */
320 binaryWriteShort(file, 0);
321 fwrite((char*) attributeCompressed, 1, attributeSize, file);
322 fwrite((char*) xCompressed, 1, xCompressedSize, file);
323 fwrite((char*) yCompressed, 1, yCompressedSize, file);
324 }
325
326 safe_free(attributeCompressed);
327 safe_free(xCompressed);
328 safe_free(yCompressed);
329 safe_free(attributeValues);
330 safe_free(xValues);
331 safe_free(yValues);
332 safe_free(decodedColors);
333 safe_free(encodedColors);
334 return 1;
335}
