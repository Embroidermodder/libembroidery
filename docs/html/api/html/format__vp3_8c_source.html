121{
122 unsigned char magicString[5];
123 unsigned char some;
124 unsigned char* softwareVendorString = 0;
125 /* unsigned char v2, ..., v18; */
126 unsigned char* anotherSoftwareVendorString = 0;
127 int numberOfColors;
128 long colorSectionOffset;
129 unsigned char magicCode[6];
130 short someShort;
131 unsigned char someByte;
132 int bytesRemainingInFile, hoopConfigurationOffset;
133 unsigned char* fileCommentString = 0; /* some software writes used settings here */
134 unsigned char* anotherCommentString = 0;
135 int i;
136
137 fread(magicString, 1, 5, file); /* %vsm% */ /* TODO: check return value */
138 some = (char)fgetc(file); /* 0 */
139 softwareVendorString = vp3ReadString(file);
140 embInt_read(file, "someShort", &someShort, EMB_INT16_LITTLE);
141 someByte = (char)fgetc(file);
142 embInt_read(file, ".", &bytesRemainingInFile, EMB_INT32_LITTLE);
143 fileCommentString = vp3ReadString(file);
144 hoopConfigurationOffset = (int)ftell(file);
145
146 vp3ReadHoopSection(file);
147
148 anotherCommentString = vp3ReadString(file);
149
150 /* TODO: review v1 thru v18 variables and use emb_unused() if needed */
151 for (i = 0; i < 18; i++) {
152 unsigned char v1;
153 v1 = (char)fgetc(file);
154 if (emb_verbose>1) {
155 printf("v%d = %d\n", i, v1);
156 }
157 }
158
159 fread(magicCode, 1, 6, file); /* 0x78 0x78 0x55 0x55 0x01 0x00 */ /* TODO: check return value */
160
161 anotherSoftwareVendorString = vp3ReadString(file);
162
163 embInt_read(file, ".", &numberOfColors, EMB_INT16_BIG);
164 colorSectionOffset = (int)ftell(file);
165 if (emb_verbose>1) {
166 puts("LIBEMBROIDERY DEBUG");
167 puts("Format vp3");
168 printf(" some: %d\n", some);
169 printf(" softwareVendorString: %s\n", softwareVendorString);
170 printf(" someByte: %d\n", someByte);
171 printf(" bytesRemainingInFile: %d\n", bytesRemainingInFile);
172 printf(" hoop config offset: %d\n", hoopConfigurationOffset);
173 printf(" another comment string: %s\n", anotherCommentString);
174 printf(" another software vendor string: %s\n", anotherSoftwareVendorString);
175 printf(" number of colors: %d\n", numberOfColors);
176 printf(" color section offset: %ld\n", colorSectionOffset);
177 }
178
179 for (i = 0; i < numberOfColors; i++) {
180 EmbThread t;
181 char tableSize;
182 int startX, startY, offsetToNextColorX, offsetToNextColorY;
183 unsigned char* threadColorNumber, *colorName, *threadVendor;
184 int unknownThreadString, numberOfBytesInColor;
185
186 strcpy(t.catalogNumber, "");
187 strcpy(t.description, "");
188 fseek(file, colorSectionOffset, SEEK_SET);
189 printf("ERROR: format-vp3.c Color Check Byte #1: 0 == %d\n", (char)fgetc(file));
190 printf("ERROR: format-vp3.c Color Check Byte #2: 5 == %d\n", (char)fgetc(file));
191 printf("ERROR: format-vp3.c Color Check Byte #3: 0 == %d\n", (char)fgetc(file));
192 embInt_read(file, ".", &colorSectionOffset, EMB_INT32_BIG);
193 colorSectionOffset += ftell(file);
194 embInt_read(file, ".", &startX, EMB_INT32_BIG);
195 embInt_read(file, ".", &startY, EMB_INT32_BIG);
196 embPattern_addStitchAbs(pattern, startX / 1000.0, -startY / 1000.0, JUMP, 1);
197
198 tableSize = (char)fgetc(file);
199 fseek(file, 1, SEEK_CUR);
200 embColor_read(file, &(t.color), 3);
201 embPattern_addThread(pattern, t);
202 fseek(file, 6*tableSize - 1, SEEK_CUR);
203
204 threadColorNumber = vp3ReadString(file);
205 colorName = vp3ReadString(file);
206 threadVendor = vp3ReadString(file);
207
208 offsetToNextColorX = fread_int32_be(file);
209 offsetToNextColorY = fread_int32_be(file);
210
211 embInt_read(file, "unknown thread string", &unknownThreadString, EMB_INT16_BIG);
212 fseek(file, unknownThreadString, SEEK_CUR);
213 numberOfBytesInColor = fread_int32_be(file);
214 fseek(file, 0x3, SEEK_CUR);
215
216 if (emb_verbose>1) {
217 printf("number of bytes in color: %d\n", numberOfBytesInColor);
218 printf("thread color number: %s\n", threadColorNumber);
219 printf("offset to next color x: %d\n", offsetToNextColorX);
220 printf("offset to next color y: %d\n", offsetToNextColorY);
221 printf("color name: %s\n", colorName);
222 printf("thread vendor: %s\n", threadVendor);
223 printf("fileCommentString: %s\n", fileCommentString);
224 }
225
226 while (ftell(file) < colorSectionOffset - 1) {
227 int lastFilePosition = ftell(file);
228 int x = vp3Decode((char)fgetc(file));
229 int y = vp3Decode((char)fgetc(file));
230 short readIn;
231 if (x == 0x80) {
232 switch (y) {
233 case 0x00:
234 case 0x03:
235 break;
236 case 0x01:
237 embInt_read(file, ".", &readIn, EMB_INT16_BIG);
238 x = vp3DecodeInt16(readIn);
239 embInt_read(file, ".", &readIn, EMB_INT16_BIG);
240 y = vp3DecodeInt16(readIn);
241 fseek(file, 2, SEEK_CUR);
242 embPattern_addStitchRel(pattern, x/ 10.0, y / 10.0, TRIM, 1);
243 break;
244 default:
245 break;
246 }
247 } else {
248 embPattern_addStitchRel(pattern, x / 10.0, y / 10.0, NORMAL, 1);
249 }
250
251 if (ftell(file) == lastFilePosition) {
252 printf("ERROR: format-vp3.c could not read stitch block in entirety\n");
253 return 0;
254 }
255 }
256 if (i + 1 < numberOfColors) {
257 embPattern_addStitchRel(pattern, 0, 0, STOP, 1);
258 }
259 }
260 embPattern_flipVertical(pattern);
261 return 1;
262}
