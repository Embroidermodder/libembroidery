22char
23readZsk(EmbPattern* pattern, FILE* file)
24{
25 char b[3];
26 int stitchType;
27 unsigned char colorNumber;
28
29 fseek(file, 0, SEEK_END);
30 if (ftell(file) < 0x230) {
31 return 0;
32 }
33
34 fseek(file, 0x230, SEEK_SET);
35 colorNumber = fgetc(file);
36 while (colorNumber != 0) {
37 EmbThread t;
38 embColor_read(file, &(t.color), 3);
39 strcpy(t.catalogNumber, "");
40 strcpy(t.description, "");
41 embPattern_addThread(pattern, t);
42 fseek(file, 0x48, SEEK_CUR);
43 colorNumber = fgetc(file);
44 }
45 fseek(file, 0x2E, SEEK_CUR);
46
47 while (fread(b, 1, 3, file) == 3) {
48 stitchType = NORMAL;
49 if (b[0] & 0x04) {
50 b[2] = -b[2];
51 }
52 if (b[0] & 0x08) {
53 b[1] = -b[1];
54 }
55 if (b[0] & 0x02) {
56 stitchType = JUMP;
57 }
58 if (b[0] & 0x20) {
59 if (b[1] == 2) {
60 stitchType = TRIM;
61 } else if (b[1] == -1) {
62 break;
63 } else {
64 if (b[2] != 0) {
65 colorNumber = b[2];
66 }
67 /* TODO: need to determine what b[1] is used for.*/
68 stitchType = STOP;
69 embPattern_changeColor(pattern, colorNumber - 1);
70 }
71 b[1] = 0;
72 b[2] = 0;
73 }
74 embPattern_addStitchRel(pattern, b[1] / 10.0,
75 b[2] / 10.0, stitchType, 0);
76 }
77
78 return 1;
79}

82char
83writeZsk(EmbPattern* pattern, FILE* file)
84{
85 int i;
86
87 fpad(file, 0x00, 0x230);
88
89 fprintf(file, "%c", pattern->thread_list->count);
90 for (i=pattern->thread_list->count; i>0; i--) {
91 EmbThread t = pattern->thread_list->thread[i-1];
92 embColor_write(file, t.color, 3);
93 fpad(file, 0x00, 0x48);
94 fprintf(file, "%c", i-1);
95 }
96
97 fpad(file, 0x00, 0x2E);
98
99 for (i=0; i<pattern->stitch_list->count; i++) {
100 EmbStitch st;
101 unsigned char b[3];
102 st = pattern->stitch_list->stitch[i];
103 b[0] = 0;
104 b[1] = st.x;
105 b[2] = st.y;
106 if (st.flags & JUMP) {
107 b[0] |= 0x02;
108 }
109 if (st.x < 0) {
110 b[0] |= 0x08;
111 b[1] = -st.x;
112 }
113 if (st.y < 0) {
114 b[0] |= 0x04;
115 b[2] = -st.y;
116 }
117 if (st.flags & TRIM) {
118 b[0] |= 0x20;
119 b[1] = 0x00;
120 b[2] = 0x00;
121 }
122 if (st.flags & STOP) {
123 b[0] |= 0x20;
124 b[1] = 0x00;
125 b[2] = st.color;
126 }
127 if (st.flags & END) {
128 b[0] |= 0x20;
129 b[1] = 0x80;
130 b[2] = 0x00;
131 fwrite(b, 1, 3, file);
132 break;
133 }
134 fwrite(b, 1, 3, file);
135 }
136 return 1;
137}
