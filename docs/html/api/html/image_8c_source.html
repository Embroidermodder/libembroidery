27void
28writeImage(FILE* file, unsigned char image[][48])
29{
30 int i, j;
31
32 if (!file) {
33 printf("ERROR: format-pec.c writeImage(), file argument is null\n");
34 return;
35 }
36 for (i = 0; i < 38; i++) {
37 for (j = 0; j < 6; j++) {
38 int offset = j * 8;
39 unsigned char output = 0;
40 output |= (unsigned char)(image[i][offset] != 0);
41 output |= (unsigned char)(image[i][offset + 1] != (unsigned char)0) << 1;
42 output |= (unsigned char)(image[i][offset + 2] != (unsigned char)0) << 2;
43 output |= (unsigned char)(image[i][offset + 3] != (unsigned char)0) << 3;
44 output |= (unsigned char)(image[i][offset + 4] != (unsigned char)0) << 4;
45 output |= (unsigned char)(image[i][offset + 5] != (unsigned char)0) << 5;
46 output |= (unsigned char)(image[i][offset + 6] != (unsigned char)0) << 6;
47 output |= (unsigned char)(image[i][offset + 7] != (unsigned char)0) << 7;
48 fwrite(&output, 1, 1, file);
49 }
50 }
51}
