15/* Finds the normalized vector perpendicular (clockwise) to the line
16 * given by v1->v2 (normal to the line) */
17EmbVector
18embLine_normalVector(EmbLine line, int clockwise)
19{
20 EmbVector result;
21 EmbReal temp;
22 result = embVector_subtract(line.end, line.start);
23 result = embVector_normalize(result);
24 temp = result.x;
25 result.x = result.y;
26 result.y = -temp;
27 if (!clockwise) {
28 result.x = -result.x;
29 result.y = -result.y;
30 }
31 return result;
32}

43/*
44 * Finds the intersection of two lines given by v1->v2 and v3->v4
45 * and sets the value in the result variable.
46 */
47EmbVector
48embLine_intersectionPoint(EmbLine line1, EmbLine line2)
49{
50 EmbReal det, C2, C1, tolerance;
51 EmbVector vec1, vec2, result;
52 emb_error = 0;
53 vec1 = embLine_toVector(line1);
54 vec2 = embLine_toVector(line2);
55 C2 = embVector_cross(line1.start, vec1);
56 C1 = embVector_cross(line2.start, vec2);
57
58 tolerance = 1e-10;
59 det = embVector_cross(vec2, vec1);
60
61 if (fabs(det) < tolerance) {
62 emb_error = 1;
63 return result;
64 }
65 result.x = (vec2.x * C2 - vec1.x * C1) / det;
66 result.y = (vec2.y * C2 - vec1.y * C1) / det;
67 return result;
68}
