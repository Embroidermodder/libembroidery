261int
262bcfFile_read(FILE* file, bcf_file* bcfFile)
263{
264 unsigned int i, numberOfDirectoryEntriesPerSector;
265 unsigned int directorySectorToReadFrom;
266
267 bcfFile->header = bcfFileHeader_read(file);
268 if (memcmp(bcfFile->header.signature, "\xd0\xcf\x11\xe0\xa1\xb1\x1a\xe1", 8) != 0) {
269 printf("bad header signature\n");
270 printf("Failed to parse header\n");
271 return 0;
272 }
273
274 bcfFile->difat = bcf_difat_create(file, bcfFile->header.numberOfFATSectors, sectorSize(bcfFile));
275 if (haveExtraDIFATSectors(bcfFile)) {
276 parseDIFATSectors(file, bcfFile);
277 }
278
279 bcfFile->fat = bcfFileFat_create(sectorSize(bcfFile));
280 for (i = 0; i < bcfFile->header.numberOfFATSectors; ++i) {
281 unsigned int fatSectorNumber = bcfFile->difat->fatSectorEntries[i];
282 seekToSector(bcfFile, file, fatSectorNumber);
283 loadFatFromSector(bcfFile->fat, file);
284 }
285
286 numberOfDirectoryEntriesPerSector = sectorSize(bcfFile) / sizeOfDirectoryEntry;
287 bcfFile->directory = CompoundFileDirectory(numberOfDirectoryEntriesPerSector);
288 directorySectorToReadFrom = bcfFile->header.firstDirectorySectorLocation;
289 while (directorySectorToReadFrom != CompoundFileSector_EndOfChain) {
290 seekToSector(bcfFile, file, directorySectorToReadFrom);
291 readNextSector(file, bcfFile->directory);
292 directorySectorToReadFrom = bcfFile->fat->fatEntries[directorySectorToReadFrom];
293 }
294
295 return 1;
296}

306FILE*
307GetFile(bcf_file* bcfFile, FILE* file, char* fileToFind)
308{
309 int filesize, sectorSize, currentSector;
310 int sizeToWrite, currentSize, totalSectors, i, j;
311 FILE* fileOut = tmpfile();
312 bcf_directory_entry* pointer = bcfFile->directory->dirEntries;
313 while (pointer) {
314 if (strcmp(fileToFind, pointer->directoryEntryName) == 0)
315 break;
316 pointer = pointer->next;
317 }
318 filesize = pointer->streamSize;
319 sectorSize = bcfFile->difat->sectorSize;
320 currentSize = 0;
321 currentSector = pointer->startingSectorLocation;
322 totalSectors = (int)ceil((float)filesize / sectorSize);
323 for (i = 0; i < totalSectors; i++) {
324 seekToSector(bcfFile, file, currentSector);
325 sizeToWrite = filesize - currentSize;
326 if (sectorSize < sizeToWrite) {
327 sizeToWrite = sectorSize;
328 }
329 for (j=0; j<sizeToWrite; j++) {
330 char input;
331 fread(&input, 1, 1, file);
332 fwrite(&input, 1, 1, fileOut);
333 }
334 currentSize += sizeToWrite;
335 currentSector = bcfFile->fat->fatEntries[currentSector];
336 }
337 return fileOut;
338}

364{
365 unsigned int i;

389 }
390 }

414unsigned int
415readFullSector(FILE* file, bcf_file_difat* bcfFile,
416 unsigned int* difatEntriesToRead)
417{
418 unsigned int i;
419 unsigned int sectorRef;
420 unsigned int nextDifatSectorInChain;
421 unsigned int entriesToReadInThisSector = 0;
422 if (*difatEntriesToRead > entriesInDifatSector(bcfFile)) {
423 entriesToReadInThisSector = entriesInDifatSector(bcfFile);
424 *difatEntriesToRead -= entriesToReadInThisSector;
425 }
426 else {
427 entriesToReadInThisSector = *difatEntriesToRead;
428 *difatEntriesToRead = 0;
429 }
430
431 for (i = 0; i < entriesToReadInThisSector; ++i) {
432 embInt_read(file, "sectorRef", &sectorRef, EMB_INT32_LITTLE);
433 bcfFile->fatSectorEntries[bcfFile->fatSectorCount] = sectorRef;
434 bcfFile->fatSectorCount++;
435 }
436 for (i = entriesToReadInThisSector; i < entriesInDifatSector(bcfFile); ++i) {
437 embInt_read(file, "sectorRef", &sectorRef, EMB_INT32_LITTLE);
438 if (sectorRef != CompoundFileSector_FreeSector) {
439 printf("ERROR: compound-file-difat.c readFullSector(), ");
440 printf("Unexpected sector value %x at DIFAT[%d]]\n", sectorRef, i);
441 }
442 }
443 embInt_read(file, "nextDifatSectorInChain", &nextDifatSectorInChain, EMB_INT32_LITTLE);
444 return nextDifatSectorInChain;
445}

514bcf_directory_entry*
515CompoundFileDirectoryEntry(FILE* file)
516{
517 const int guidSize = 16;
518 bcf_directory_entry* dir = malloc(sizeof(bcf_directory_entry));
519 if (dir == NULL) {
520 printf("ERROR: compound-file-directory.c CompoundFileDirectoryEntry(), cannot allocate memory for dir\n");
521 return 0;
522 }
523 memset(dir->directoryEntryName, 0, 32);
524 parseDirectoryEntryName(file, dir);
525 dir->next = 0;
526 dir->directoryEntryNameLength = fread_uint16(file);
527 dir->objectType = (unsigned char)fgetc(file);
528 if ((dir->objectType != ObjectTypeStorage) && (dir->objectType != ObjectTypeStream) && (dir->objectType != ObjectTypeRootEntry)) {
529 printf("ERROR: compound-file-directory.c CompoundFileDirectoryEntry()");
530 printf(", unexpected object type: %d\n", dir->objectType);
531 return 0;
532 }
533 dir->colorFlag = (unsigned char)fgetc(file);
534 embInt_read(file, "left sibling id", &(dir->leftSiblingId), EMB_INT32_LITTLE);
535 embInt_read(file, "right sibling id", &(dir->rightSiblingId), EMB_INT32_LITTLE);
536 embInt_read(file, "child id", &(dir->childId), EMB_INT32_LITTLE);
537 fread(dir->CLSID, 1, guidSize, file);
538 embInt_read(file, "state bits", &(dir->stateBits), EMB_INT32_LITTLE);
539 dir->creationTime = parseTime(file);
540 dir->modifiedTime = parseTime(file);
541 embInt_read(file, "starting sector location", &(dir->startingSectorLocation), EMB_INT32_LITTLE);
542 /* StreamSize should really be __int64 or long long,
543 * but for our uses we should never run into an issue */
544 embInt_read(file, "dir->streamSize", &(dir->streamSize), EMB_INT32_LITTLE);
545 /* top portion of int64 */
546 embInt_read(file, "dir->streamSizeHigh", &(dir->streamSizeHigh), EMB_INT32_LITTLE);
547 return dir;
548}

650void
651embSatinOutline_generateSatinOutline(EmbArray *lines, EmbReal thickness, EmbSatinOutline* result)
652{
653 int i;
654 EmbLine line1, line2;
655 EmbSatinOutline outline;
656 EmbVector out;
657 EmbVector v1;
658 EmbVector temp;
659 EmbLine line;
660
661 EmbReal halfThickness = thickness / 2.0;
662 int intermediateOutlineCount = 2 * lines->count - 2;
663 outline.side1 = embArray_create(EMB_VECTOR);
664 if (!outline.side1) {
665 printf("ERROR: emb-satin-line.c embSatinOutline_generateSatinOutline(), cannot allocate memory for outline->side1\n");
666 return;
667 }
668 outline.side2 = embArray_create(EMB_VECTOR);
669 if (!outline.side2) {
670 printf("ERROR: emb-satin-line.c embSatinOutline_generateSatinOutline(), cannot allocate memory for outline->side2\n");
671 return;
672 }
673
674 for (i = 1; i < lines->count; i++) {
675 line.start = lines->geometry[i - 1].object.vector;
676 line.end = lines->geometry[i].object.vector;
677
678 v1 = embLine_normalVector(line, 1);
679
680 temp = embVector_scale(v1, halfThickness);
681 temp = embVector_add(temp, lines->geometry[i-1].object.vector);
682 embArray_addVector(outline.side1, temp);
683 temp = embVector_add(temp, lines->geometry[i].object.vector);
684 embArray_addVector(outline.side1, temp);
685
686 temp = embVector_scale(v1, -halfThickness);
687 temp = embVector_add(temp, lines->geometry[i - 1].object.vector);
688 embArray_addVector(outline.side2, temp);
689 temp = embVector_add(temp, lines->geometry[i].object.vector);
690 embArray_addVector(outline.side2, temp);
691 }
692
693 if (!result) {
694 printf("ERROR: emb-satin-line.c embSatinOutline_generateSatinOutline(), result argument is null\n");
695 return;
696 }
697 result->side1 = embArray_create(EMB_VECTOR);
698 if (!result->side1) {
699 printf("ERROR: emb-satin-line.c embSatinOutline_generateSatinOutline(), cannot allocate memory for result->side1\n");
700 return;
701 }
702 result->side2 = embArray_create(EMB_VECTOR);
703 if (!result->side2) {
704 printf("ERROR: emb-satin-line.c embSatinOutline_generateSatinOutline(), cannot allocate memory for result->side2\n");
705 return;
706 }
707
708 embArray_addVector(result->side1, outline.side1->geometry[0].object.vector);
709 embArray_addVector(result->side2, outline.side2->geometry[0].object.vector);
710
711 for (i = 3; i < intermediateOutlineCount; i += 2) {
712 line1.start = outline.side1->geometry[i - 3].object.vector;
713 line1.end = outline.side1->geometry[i - 2].object.vector;
714 line2.start = outline.side1->geometry[i - 1].object.vector;
715 line2.end = outline.side1->geometry[i].object.vector;
716 out = embLine_intersectionPoint(line1, line2);
717 if (emb_error) {
718 puts("No intersection point.");
719 }
720 embArray_addVector(result->side1, out);
721
722 line1.start = outline.side2->geometry[i - 3].object.vector;
723 line1.end = outline.side2->geometry[i - 2].object.vector;
724 line2.start = outline.side2->geometry[i - 1].object.vector;
725 line2.end = outline.side2->geometry[i].object.vector;
726 out = embLine_intersectionPoint(line1, line2);
727 if (emb_error) {
728 puts("No intersection point.");
729 }
730 embArray_addVector(result->side2, out);
731 }
732
733 embArray_addVector(result->side1, outline.side1->geometry[2 * lines->count - 3].object.vector);
734 embArray_addVector(result->side2, outline.side2->geometry[2 * lines->count - 3].object.vector);
735 result->length = lines->count;
736}

745EmbArray*
746embSatinOutline_renderStitches(EmbSatinOutline* result, EmbReal density)
747{
748 int i, j;
749 EmbVector currTop, currBottom, topDiff, bottomDiff, midDiff;
750 EmbVector midLeft, midRight, topStep, bottomStep;
751 EmbArray* stitches = 0;
752 int numberOfSteps;
753 EmbReal midLength;
754
755 if (!result) {
756 printf("ERROR: emb-satin-line.c embSatinOutline_renderStitches(), result argument is null\n");
757 return 0;
758 }
759
760 if (result->length > 0) {
761 for (j = 0; j < result->length - 1; j++) {
762 EmbGeometry *g10 = &(result->side1->geometry[j+0]);
763 EmbGeometry *g11 = &(result->side1->geometry[j+1]);
764 EmbGeometry *g20 = &(result->side2->geometry[j+0]);
765 EmbGeometry *g21 = &(result->side2->geometry[j+1]);
766 topDiff = embVector_subtract(g10->object.vector, g11->object.vector);
767 bottomDiff = embVector_subtract(g21->object.vector, g20->object.vector);
768
769 midLeft = embVector_average(g10->object.vector, g20->object.vector);
770 midRight = embVector_average(g11->object.vector, g21->object.vector);
771
772 midDiff = embVector_subtract(midLeft, midRight);
773 midLength = embVector_length(midDiff);
774
775 numberOfSteps = (int)(midLength * density / 200);
776 topStep = embVector_scale(topDiff, 1.0/numberOfSteps);
777 bottomStep = embVector_scale(bottomDiff, 1.0/numberOfSteps);
778 currTop = g10->object.vector;
779 currBottom = g20->object.vector;
780
781 for (i = 0; i < numberOfSteps; i++) {
782 if (!stitches) {
