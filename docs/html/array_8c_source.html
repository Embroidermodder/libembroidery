18EmbArray*
19embArray_create(int type)
20{
21 EmbArray *a;
22 a = (EmbArray*)malloc(sizeof(EmbArray));
23 a->type = type;
24 a->length = CHUNK_SIZE;
25 a->count = 0;
26 switch (type) {
27 case EMB_STITCH:
28 a->stitch = (EmbStitch*)malloc(CHUNK_SIZE*sizeof(EmbStitch));
29 break;
30 case EMB_THREAD:
31 a->thread = (EmbThread*)malloc(CHUNK_SIZE*sizeof(EmbThread));
32 break;
33 default:
34 a->geometry = (EmbGeometry*)malloc(CHUNK_SIZE*sizeof(EmbGeometry));
35 break;
36 }
37 return a;
38}

40int
41embArray_resize(EmbArray *a)
42{
43 if (a->count < a->length) {
44 return 1;
45 }
46 a->length += CHUNK_SIZE;
47 switch (a->type) {
48 case EMB_STITCH:
49 a->stitch = (EmbStitch*)realloc(a->stitch, a->length*sizeof(EmbStitch));
50 if (!a->stitch) {
51 /* TODO: Error reporting */
52 return 0;
53 }
54 break;
55 case EMB_THREAD:
56 a->thread = (EmbThread*)realloc(a->thread, a->length*sizeof(EmbThread));
57 if (!a->thread) {
58 /* TODO: Error reporting */
59 return 0;
60 }
61 break;
62 default:
63 a->geometry = (EmbGeometry *)realloc(a->geometry, a->length*sizeof(EmbGeometry));
64 if (!a->geometry) {
65 /* TODO: Error reporting */
66 return 0;
67 }
68 break;
69 }
70
71 return 1;
72}

74void embArray_copy(EmbArray *dst, EmbArray *src)
75{
76 dst = embArray_create(src->type);
77 dst->length = src->length;
78 dst->count = src->count;
79 embArray_resize(dst);
80 /* BUG: Potential failure to copy path memory, only copy pointers? */
81
82 switch (dst->type) {
83 case EMB_STITCH:
84 memcpy(dst->stitch, src->stitch, sizeof(EmbStitch)*src->count);
85 break;
86 case EMB_THREAD:
87 memcpy(dst->thread, src->thread, sizeof(EmbThread)*src->count);
88 break;
89 default:
90 memcpy(dst->geometry, src->geometry, sizeof(EmbGeometry)*src->count);
91 break;
92 }
93}

238void
239embArray_free(EmbArray* a)
240{
241 int i;
242 if (!a) {
243 return;
244 }
245 switch (a->type) {
246 case EMB_STITCH:
247 free(a->stitch);
248 break;
249 case EMB_THREAD:
250 free(a->thread);
251 break;
252 default:
253 for (i = 0; i < a->count; i++) {
254 EmbGeometry g = a->geometry[i];
255 switch (a->geometry[i].type) {
256 case EMB_PATH: {
257 embArray_free(g.object.path.pointList);
258 break;
259 }
260 case EMB_POLYGON: {
261 embArray_free(g.object.polygon.pointList);
262 break;
263 }
264 case EMB_POLYLINE: {
265 embArray_free(g.object.polyline.pointList);
266 break;
267 }
268 default:
269 break;
270 }
271 }
272 free(a->geometry);
273 break;
274 }
275 free(a);
276}
