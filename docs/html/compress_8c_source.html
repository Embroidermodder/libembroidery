40/* This avoids the now unnecessary compression by placing a
41 * minimal header of 6 bytes and using only literals in the
42 * huffman compressed part (see the sources above).
43 */
44int hus_compress(char *data, int length, char *output, int *output_length)
45{
46 unsigned char *a = (unsigned char*)output;
47 a[0] = length%256;
48 a[1] = (length/256)%256;
49 a[2] = 0x02;
50 a[3] = 0xA0;
51 a[4] = 0x01;
52 a[5] = 0xFE;
53 memcpy(output+6, data, length);
54 *output_length = length+6;
55 return 0;
56}

60void huffman_build_table(huffman *h)
61{
62 int bit_length, i, max_length, size;
63 max_length = 0;
64 size = 1 << h->table_width;
65 for (i = 0; i < h->table_width; i++) {
66 if (h->lengths[i] > max_length) {
67 max_length = h->lengths[i];
68 }
69 }
70 for (bit_length=1; bit_length<=h->table_width; bit_length++) {
71 int j;
72 size /= 2;
73 for (j=0; j < h->nlengths; j++) {
74 if (h->lengths[j] == bit_length) {
75 int k;
76 for (k=0; k<size; k++) {
77 h->table[h->ntable+k] = j;
78 h->ntable++;
79 }
80 }
81 }
82 }
83}

87int *huffman_lookup(huffman h, int byte_lookup)
88{
89 int *out = huffman_lookup_data;
90 if (h.table_width == 0) {
91 out[0] = h.default_value;
92 out[1] = 0;
93 return out;
94 }
95 out[0] = h.table[byte_lookup >> (16-h.table_width)];
96 out[1] = h.lengths[out[0]];
97 return out;
98}

106int compress_get_bits(compress *c, int length)
107{
108 int i, end_pos_in_bits, start_pos_in_bytes,
109 end_pos_in_bytes, value, mask_sample_bits,
110 unused_bits, original;
111
112 end_pos_in_bits = c->bit_position + length - 1;
113 start_pos_in_bytes = c->bit_position / 8;
114 end_pos_in_bytes = end_pos_in_bits / 8;
115 value = 0;
116
117 for (i=start_pos_in_bytes; i < end_pos_in_bytes+1; i++) {
118 value <<= 8;
119 if (i > c->input_length) {
120 break;
121 }
122 value |= (c->input_data[i]) & 0xFF;
123 }
124
125 unused_bits = (7 - end_pos_in_bits) % 8;
126 mask_sample_bits = (1<<length) - 1;
127 original = (value >> unused_bits) & mask_sample_bits;
128 return original;
129}

143int compress_read_variable_length(compress *c)
144{
145 int q, m, s;
146 m = compress_pop(c, 3);
147 if (m!=7) {
148 return m;
149 }
150 for (q=0; q<13; q++) {
151 s = compress_pop(c, 1);
152 if (s) {
153 m++;
154 }
155 else {
156 break;
157 }
158 }
159 return m;
160}

162void compress_load_character_length_huffman(compress *c)
163{
164 int count;
165 count = compress_pop(c, 5);
166 if (count == 0) {
167 c->character_length_huffman.default_value = compress_pop(c, 5);
168 }
169 else {
170 int i;
171 for (i = 0; i < count; i++) {
172 c->character_length_huffman.lengths[i] = 0;
173 }
174 for (i = 0; i < count; i++) {
175 if (i==3) {
176 i += compress_pop(c, 2);
177 }
178 c->character_length_huffman.lengths[i] = compress_read_variable_length(c);
179 }
180 }
181 huffman_build_table(&(c->character_length_huffman));
182}

184void compress_load_character_huffman(compress *c)
185{
186 int count;
187 count = compress_pop(c, 9);
188 if (count == 0) {
189 c->character_huffman.default_value = compress_pop(c, 9);
190 }
191 else {
192 int i = 0;
193 while (i < count) {
194 int *h = huffman_lookup(c->character_length_huffman,
195 compress_peek(c, 16));
196 c->bit_position += h[1];
197 if (h[0]==0) {
198 i += h[0];
199 }
200 else if (h[0]==1) {
201 i += 3 + compress_pop(c, 4);
202 }
203 else if (h[0]==2) {
204 i += 20 + compress_pop(c, 9);
205 }
206 else {
207 c->character_huffman.lengths[i] = h[0] - 2;
208 i++;
209 }
210 }
211 }
212 huffman_build_table(&(c->character_huffman));
213}

239int compress_get_token(compress *c)
240{
241 int *h;
242 if (c->block_elements <= 0) {
243 compress_load_block(c);
244 }
245 c->block_elements--;
246 h = huffman_lookup(c->character_huffman, compress_peek(c, 16));
247 c->bit_position += h[1];
248 return h[0];
249}

251int compress_get_position(compress *c)
252{
253 int *h, v;
254 h = huffman_lookup(c->distance_huffman, compress_peek(c, 16));
255 c->bit_position += h[1];
256 if (h[0] == 0) {
257 return 0;
258 }
259 v = h[0] - 1;
260 v = (1<<v) + compress_pop(c, v);
261 return v;
262}

264int hus_decompress(char *data, int length, char *output, int *output_length)
265{
266 int character, i, j;
267 compress *c = (compress*)malloc(sizeof(compress));
268 c->bit_position = 0;
269 c->input_data = data;
270 c->input_length = length;
271 c->bits_total = length*8;
272 i = 0;
273 while (c->bits_total > c->bit_position && i < *output_length) {
274 /* process token */
275 character = 0; /* fix this */
276 if (character < 0x100) {
277 output[i] = (char)character;
278 i++;
279 }
280 else if (character == 510) {
281 break;
282 }
283 else {
284 length = character - 253;
285 /* not sure about i here */
286 c->bit_position = i - compress_get_position(c) - 1;
287 for (j=c->bit_position; j < c->bit_position+length; j++) {
288 output[i] = output[j];
289 i++;
290 }
291 }
292 }
293 free(c);
294 return 0;
295}
