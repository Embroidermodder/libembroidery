25void
26writeImage(FILE* file, unsigned char image[][48])
27{
28 int i, j;
29
30 if (!file) {
31 printf("ERROR: format-pec.c writeImage(), file argument is null\n");
32 return;
33 }
34 for (i = 0; i < 38; i++) {
35 for (j = 0; j < 6; j++) {
36 int offset = j * 8;
37 unsigned char output = 0;
38 output |= (unsigned char)(image[i][offset] != 0);
39 output |= (unsigned char)(image[i][offset + 1] != (unsigned char)0) << 1;
40 output |= (unsigned char)(image[i][offset + 2] != (unsigned char)0) << 2;
41 output |= (unsigned char)(image[i][offset + 3] != (unsigned char)0) << 3;
42 output |= (unsigned char)(image[i][offset + 4] != (unsigned char)0) << 4;
43 output |= (unsigned char)(image[i][offset + 5] != (unsigned char)0) << 5;
44 output |= (unsigned char)(image[i][offset + 6] != (unsigned char)0) << 6;
45 output |= (unsigned char)(image[i][offset + 7] != (unsigned char)0) << 7;
46 fwrite(&output, 1, 1, file);
47 }
48 }
49}

80/* Basic Render
81 * ------------
82 * Backends rendering to nanosvg/stb_image.
83 *
84 * The caller is responsible for the memory in p.
85 */
86int
87embPattern_render(EmbPattern *p, char *fname)
88{
89 const char *tmp_fname = "libembroidery_temp.svg";
90 NSVGimage *image = NULL;
91 NSVGrasterizer rast;
92 unsigned char *img_data = NULL;
93 embPattern_writeAuto(p, tmp_fname);
94 image = nsvgParseFromFile(tmp_fname, "px", 96.0f);
95 img_data = malloc(4*image->width*image->height);
96 nsvgRasterize(
97 &rast,
98 image,
99 0, 0, 1,
100 img_data,
101 image->width,
102 image->height,
103 4*image->width);
104 stbi_write_png(
105 fname,
106 image->width,
107 image->height,
108 4,
109 img_data,
110 4*image->width);
111 return 0;
112}
